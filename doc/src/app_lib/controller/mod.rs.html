28pub async fn get_inc_data(
29    Path(editor_name): Path<String>
30) -> ResponseResult<Option<Arc<Operations>>> {
31    let editor = ContextHelper::get_editor(&editor_name);
32    if editor.is_none() {
33        return Err(AppError(anyhow::anyhow!("工程项目不存在".to_string())));
34    }
35    let editor = editor.unwrap();
36    let manager = editor.get_state().await.resource_manager();
37    let operations =
38        manager.resource_table.take::<Operations>("inc_data".to_string());
39    res!(operations)
40}

42/// 获取数据树
43pub async fn get_data_tree(
44    Json(param): Json<GetDataTreeRequest>
45) -> ResponseResult<GcxmTreeItem> {
46    let editor = ContextHelper::get_editor(&param.editor_name);
47    if editor.is_none() {
48        return Err(AppError(anyhow::anyhow!("工程项目不存在".to_string())));
49    }
50    let editor = editor.unwrap();
51    let doc = editor.doc().await;
52    let node = doc.get_node(&param.id);
53    if node.is_none() {
54        return Err(AppError(anyhow::anyhow!("节点不存在".to_string())));
55    }
56    let node = node.unwrap();
57    let mut nodes: Vec<Arc<Node>> = doc.descendants(&param.id);
58    nodes.push(node);
59    let parent_map = &doc.get_inner().parent_map;
60    if let Some(root_item) =
61        GcxmTreeItem::from_nodes(param.id.clone(), nodes, parent_map)
62    {
63        res!(root_item)
64    } else {
65        Err(AppError(anyhow::anyhow!("无法构建工程树,未找到根节点")))
66    }
67}

74/// 获取历史记录
75pub async fn get_history(
76    Json(param): Json<GetHistoryVersionCammand>
77) -> ResponseResult<Vec<HistoryEntry>> {
78    let editor = ContextHelper::get_editor(&param.editor_name);
79    if editor.is_none() {
80        return Err(AppError(anyhow::anyhow!("工程项目不存在".to_string())));
81    }
82    let editor = editor.unwrap();
83    let history_manager = editor.get_history_manager().await;
84    if history_manager.is_none() {
85        return Err(AppError(anyhow::anyhow!("历史记录不存在".to_string())));
86    }
87    let history_manager = history_manager.unwrap();
88    let history_version = history_manager.get_history();
89    let history = history_version;
90    let mut history_result = history
91        .past
92        .iter()
93        .map(render_history_entry)
94        .collect::<Vec<HistoryEntry>>();
95    history_result.push(render_history_entry(&history.present));
96    let history_future = history
97        .future
98        .iter()
99        .map(render_history_entry)
100        .collect::<Vec<HistoryEntry>>();
101    history_result.extend(history_future);
102    res!(history_result)
103}

105/// 渲染历史记录描述
106fn render_history_entry(item: &HistoryEntryWithMeta) -> HistoryEntry {
107    let description =
108        render(&item.description, item.meta.clone().into()).unwrap();
109    // 日期格式化成yyyy-MM-dd HH:mm:ss
110    let timestamp = DateTime::<Local>::from(item.timestamp)
111        .format("%Y-%m-%d %H:%M:%S")
112        .to_string();
113    return HistoryEntry {
114        current: false,
115        state_version: item.state.version,
116        description: description.to_string(),
117        timestamp: timestamp,
118    };
119}

134#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
135pub struct GcxmTreeItem {
136    pub id: NodeId,
137    pub r#type: String,
138    pub attrs: Attrs,
139    pub children: Vec<GcxmTreeItem>,
140    pub marks: Vec<Mark>,
141}
142
143impl GcxmTreeItem {
144    fn from_nodes(
145        root_id: NodeId,
146        nodes: Vec<Arc<Node>>,
147        parent_map: &im::HashMap<NodeId, NodeId>,
148    ) -> Option<Self> {
149        use std::collections::HashMap;
150        if nodes.is_empty() {
151            return None;
152        }
153        let node_map: HashMap<NodeId, Arc<Node>> =
154            nodes.iter().map(|n| (n.id.clone(), n.clone())).collect();
155
156        fn build_tree(
157            id: &NodeId,
158            node_map: &HashMap<NodeId, Arc<Node>>,
159            parent_map: &im::HashMap<NodeId, NodeId>,
160        ) -> Option<GcxmTreeItem> {
161            let node = node_map.get(id)?;
162            let children: Vec<GcxmTreeItem> = node_map
163                .iter()
164                .filter(|(_, n)| parent_map.get(&n.id) == Some(id))
165                .filter_map(|(cid, _)| build_tree(cid, node_map, parent_map))
166                .collect();
167            Some(GcxmTreeItem {
168                id: node.id.clone(),
169                r#type: node.r#type.to_string(),
170                attrs: node.attrs.clone(),
171                children,
172                marks: node.marks.iter().cloned().collect(),
173            })
174        }
175
176        build_tree(&root_id, &node_map, parent_map)
177    }
178}
