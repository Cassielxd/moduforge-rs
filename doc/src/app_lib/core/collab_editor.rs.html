127impl CollabEditor {
128    pub async fn create(options: CollabEditorOptions) -> ForgeResult<Self> {
129        // 创建协作同步管理器
130        let mut sync_manager = CollabSyncManager::new(&options).await?;
131        // 订阅同步事件
132        let mut event_rx =
133            sync_manager.provider.subscribe_sync_events().unwrap();
134        // 启动协作同步
135        sync_manager.start().await;
136        // 等待初始化完成
137        // 设置超时 10 秒
138        let timeout = tokio::time::timeout(Duration::from_secs(10), async {
139            while let Ok(event) = event_rx.recv().await {
140                if let SyncEvent::InitialSyncCompleted { has_data, .. } = event
141                {
142                    return Ok(has_data);
143                }
144            }
145            Err(ForgeError::Internal {
146                message: "服务器异常，请检查服务器是否启动".to_string(),
147                location: None,
148            })
149        })
150        .await
151        .map_err(|e| ForgeError::Internal {
152            message: "服务器异常，请检查服务器是否启动".to_string(),
153            location: None,
154        })?;
155        // 创建协作扩展
156        let mut ext = Extension::new();
157        ext.add_plugin({
158            Arc::new(Plugin::new(PluginSpec {
159                state_field: Some(Arc::new(CollabStateField::new(
160                    sync_manager.awareness.clone(),
161                ))),
162                tr: Arc::new(CollabPlugin),
163            }))
164        });
165        // 添加协作扩展
166        let mut options = options;
167        options.editor_options =
168            options.editor_options.add_extension(Extensions::E(ext));
169        // 创建编辑器
170        let editor = match timeout {
171            Ok(true) => {
172                //有数据 反向同步数据 并创建 编辑器
173                let awareness_lock = sync_manager.awareness.read().await;
174                //把 远程的 文档 转换成 树
175                match Utils::apply_yrs_to_tree(awareness_lock.doc()) {
176                    Ok(tree) => {
177                        println!("有数据 反向同步数据 并创建 编辑器");
178                        //转换成功
179                        let pool =
180                            NodePool::new(Arc::new(tree)).as_ref().clone();
181                        let options = options
182                            .editor_options
183                            .clone()
184                            .set_content(Content::NodePool(pool));
185                        ForgeAsyncRuntime::create(options).await?
186                    },
187                    Err(_) => {
188                        println!(
189                            "转换失败 创建一个本地编辑器 并同步数据到远程1"
190                        );
191                        //转换失败 创建一个本地编辑器 并同步数据到远程
192                        let ed = ForgeAsyncRuntime::create(
193                            options.editor_options.clone(),
194                        )
195                        .await?;
196                        sync_manager.sync_to_remote(&ed).await?;
197                        ed
198                    },
199                }
200            },
201
202            Ok(false) => {
203                println!("没有数据 创建一个本地编辑器 并同步数据到远程2");
204                //没有数据 创建一个本地编辑器 并同步数据到远程
205                let ed =
206                    ForgeAsyncRuntime::create(options.editor_options.clone())
207                        .await?;
208                sync_manager.sync_to_remote(&ed).await?;
209                ed
210            },
211            _ => {
212                return Err(ForgeError::Other(anyhow::anyhow!(
213                    "服务器异常，请检查服务器是否启动"
214                )));
215            },
216        };
217
218        let editor_arc = Arc::new(RwLock::new(editor));
219
220        let mut collab_editor = Self {
221            editor: editor_arc.clone(),
222            options,
223            sync_manager: Arc::new(sync_manager),
224        };
225
226        // 启动同步管理器，传入编辑器引用
227        if let Err(e) = Arc::get_mut(&mut collab_editor.sync_manager)
228            .unwrap()
229            .start_with_editor(editor_arc)
230            .await
231        {
232            eprintln!("启动同步管理器失败: {}", e);
233        }
234
235        Ok(collab_editor)
236    }

305#[derive(Debug, Clone, PartialEq)]
306pub enum ChangeType {
307    /// Determines a change that resulted in adding a consecutive number of new elements:
308    /// - For [Array] it's a range of inserted elements.
309    /// - For [XmlElement] it's a range of inserted child XML nodes.
310    Added(Vec<Value>),
311    Removed(u32),
312    Retain(u32),
313}
314#[derive(Debug, Clone, PartialEq)]
315pub enum EntryChangeType {
316    /// 通知在指定条目下插入了一个新值。.
317    Inserted(Value),
318
319    /// Informs about a change of old value (1st field) to a new one (2nd field) under
320    /// a corresponding entry.
321    Updated(Value, Value),
322
323    /// Informs about a removal of a corresponding entry - contains a removed value.
324    Removed(Value),
325}

327/// 同步事件类型
328#[derive(Debug, Clone)]
329pub enum SyncEventType {
330    /// 来自本地的事务
331    ArrayChange(Vec<Value>, Vec<ChangeType>),
332    /// yrs 深度变化事件
333    MapChange(Vec<Value>, Vec<EntryChangeType>), // 简化为字节数组，避免复杂的类型问题
334}

352    pub async fn new(options: &CollabEditorOptions) -> ForgeResult<Self> {
353        let doc = Doc::new();
354        let awareness: AwarenessRef =
355            Arc::new(tokio::sync::RwLock::new(Awareness::new(doc)));

361                awareness.clone(),
362            )
363            .await,
364            awareness,
365            event_sender: None,

369    /// 启动同步管理器，设置监听器和事件处理
370    pub async fn start_with_editor(
371        &mut self,
372        editor: Arc<RwLock<ForgeAsyncRuntime>>,
373    ) -> ForgeResult<()> {
374        // 连接到服务器
375        self.provider.connect().await;
376
377        // 创建事件通道
378        let (sender, mut receiver) = mpsc::unbounded_channel();
379        self.event_sender = Some(sender.clone());
380
381        // 设置 yrs 深度监听器
382        self.setup_yrs_listener(sender.clone()).await?;
383
384        // 启动事件处理循环
385        let awareness_clone = self.awareness.clone();
386        let editor_clone = editor.clone();
387        tokio::spawn(async move {
388            while let Some(event) = receiver.recv().await {
389                if let Err(e) = Self::handle_sync_event(
390                    event,
391                    editor_clone.clone(),
392                    awareness_clone.clone(),
393                )
394                .await
395                {
396                    eprintln!("处理同步事件时出错: {}", e);
397                }
398            }
399        });
400
401        println!("协作同步管理器已启动");
402        Ok(())
403    }

405    /// 设置 yrs nodes 监听器
406    /// 专门监听 nodes 相关的变化，而不是整个文档
407    async fn setup_yrs_listener(
408        &mut self,
409        sender: mpsc::UnboundedSender<Vec<SyncEventType>>,
410    ) -> ForgeResult<()> {
411        let awareness_lock = self.awareness.read().await;
412        let doc = awareness_lock.doc();
413
414        // 获取或创建 nodes 映射
415        let nodes_map = doc.get_or_insert_map("nodes");
416
417        // 使用 observe_deep 监听 nodes 映射的深度变化
418        let sender_clone = sender.clone();
419        self.provider.subscription(nodes_map.observe_deep(move |_txn, events| {
420            let sender = sender_clone.clone();
421            let mut event_vec = Vec::new();
422            // 检查是否有事件（使用 iter() 方法）
423            let event_count = events.iter().count();
424            for event in events.iter() {
425                //path 转换成数组
426                match event {
427                    mf_collab_client::yrs::types::Event::Array(array_event) => {
428                        let path = array_event.path().iter().map(|path| serde_json::to_value(path).unwrap()).collect::<Vec<serde_json::Value>>();
429                        array_event.delta(_txn).iter().for_each(|delta| {
430                            // 将 delta 转换为 ChangeType
431                            let change_type = match delta {
432                                Change::Added(values) => ChangeType::Added(values.iter()
433                                .filter_map(|value| {
434                                    if let mf_collab_client::yrs::Value::Any(ref any) = value {
435                                        Utils::yrs_any_to_json_value(any)
436                                    } else {
437                                        None
438                                    }
439                                })
440                                .collect()),
441                                Change::Removed(count) => ChangeType::Removed(*count),
442                                Change::Retain(count) => ChangeType::Retain(*count),
443                            };
444                            event_vec.push(SyncEventType::ArrayChange(path.clone(), vec![change_type]));
445                        });
446                    }
447                    mf_collab_client::yrs::types::Event::Map(map_event) => {
448                        map_event.keys(_txn).iter().for_each(|(key, value)| {
449                            let path = vec![serde_json::to_value(key).unwrap()];
450                            let change_type = match value {
451                                EntryChange::Inserted(value) => {
452                                    if let mf_collab_client::yrs::Value::Any(ref any) = value {
453                                        EntryChangeType::Inserted(Utils::yrs_any_to_json_value(any).unwrap())
454                                    } else {
455                                        // 处理非 Any 类型（可选：用默认值或跳过）
456                                        return;
457                                    }
458                                }
459                                EntryChange::Updated(value, value1) => {
460                                    if let (mf_collab_client::yrs::Value::Any(ref any0), mf_collab_client::yrs::Value::Any(ref any1)) = (value, value1) {
461                                        EntryChangeType::Updated(
462                                            Utils::yrs_any_to_json_value(any0).unwrap(),
463                                            Utils::yrs_any_to_json_value(any1).unwrap()
464                                        )
465                                    } else {
466                                        return;
467                                    }
468                                }
469                                EntryChange::Removed(value) => {
470                                    if let mf_collab_client::yrs::Value::Any(ref any) = value {
471                                        EntryChangeType::Removed(Utils::yrs_any_to_json_value(any).unwrap())
472                                    } else {
473                                        return;
474                                    }
475                                }
476                            };
477                            event_vec.push(SyncEventType::MapChange(path, vec![change_type]));
478                        });
479                    }
480                    _ => {
481                    }
482                }
483            }
484            if event_vec.len() > 0 {
485                println!("检测到 nodes 深度变化: {} 个事件", event_count);
486
487                // 简化事件处理，只发送事件计数
488                if let Err(e) = sender.send(event_vec) {
489                    eprintln!("发送 nodes 深度变化事件失败: {}", e);
490                }
491            }
492        }));
493
494        println!("yrs nodes 监听器已设置");
495        Ok(())
496    }

498    /// 处理同步事件
499    async fn handle_sync_event(
500        events: Vec<SyncEventType>,
501        _editor: Arc<RwLock<ForgeAsyncRuntime>>,
502        _awareness: AwarenessRef,
503    ) -> ForgeResult<()> {
504        for event in events {
505            match event {
506                SyncEventType::ArrayChange(_path, changes) => {
507                    // path 数组第一个对应的 是 节点id
508                    for change in changes {
509                        match change {
510                            ChangeType::Added(values) => {
511                                println!("处理 Added 事件: {:?}", values);
512                                // 转换成mark
513                            },
514                            ChangeType::Removed(index) => {
515                                println!("处理 Removed 事件: {:?}", index);
516                            },
517                            ChangeType::Retain(index) => {
518                                println!("处理 Retain 事件: {:?}", index);
519                            },
520                        }
521                    }
522                },
523                SyncEventType::MapChange(path, changes) => {
524                    for change in changes {
525                        match change {
526                            EntryChangeType::Inserted(value) => {
527                                println!("处理 Inserted 事件: {:?}", value);
528                            },
529                            EntryChangeType::Updated(value, value1) => {
530                                println!("处理 Updated 事件: {:?}", value);
531                            },
532                            EntryChangeType::Removed(value) => {
533                                println!("处理 Removed 事件: {:?}", value);
534                            },
535                        }
536                    }
537                },
538            }
539        }
540
541        Ok(())
542    }
