26pub fn init_fbfx_csxm_fields() -> Vec<Node> {
27    let mut fb = FB.clone();
28    fb.set_attrs(get_attr_spec());
29    let mut qd = QD.clone();
30    qd.set_attrs(get_attr_spec());
31    let mut de = DE.clone();
32    de.set_attrs(get_attr_spec());
33    let mut rcj = RCJ.clone();
34    rcj.set_attrs(get_attr_spec());
35    let mut fbfx = FBFX.clone();
36    fbfx.set_attrs(get_attr_name("分部分项"));
37    let mut csxm = CSXM.clone();
38    csxm.set_attrs(get_attr_name("措施项目"));
39    vec![fb, qd, de, rcj, fbfx, csxm]
40}
41fn get_attr_name(name: &str) -> HashMap<String, AttributeSpec> {
42    let mut att = HashMap::new();
43    att.insert(
44        "name".to_string(),
45        AttributeSpec { default: Some(name.into()) },
46    );
47    att
48}
