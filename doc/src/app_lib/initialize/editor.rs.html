38//获取编辑器配置
39pub async fn init_collab_options(
40    create_callback: Arc<dyn NodePoolFnTrait>,
41    room_name: String,
42) -> CollabEditorOptions {
43    let mut builder = EditorOptionsBuilder::new();
44    builder = builder
45        .content(Content::NodePoolFn(create_callback))
46        // 设置历史记录限制
47        .history_limit(20)
48        // 添加扩展
49        .extensions(init_extension())
50        // 添加中间件
51        .add_middleware(
52            middleware::collect_fbfx_csxm::CollectFbfxCsxmMiddleware,
53        );
54    let options = builder.build();
55    CollabEditorOptions {
56        editor_options: options,
57        server_url: "ws://127.0.0.1:8080/collaboration".to_string(),
58        room_name: room_name,
59    }
60}

80//获取扩展
81pub fn init_extension() -> Vec<Extensions> {
82    let mut extensions = vec![
83        Extensions::M(marks::BG_COLOR.clone()),
84        Extensions::M(marks::FOOTNOTE.clone()),
85    ];
86    // 工程项目、单项、单位Node
87    let nodes = init_project_structure();
88    for mut node in nodes {
89        if node.get_name() == DWGC_STR {
90            node.set_content(&format!("{}|{}+", FBFX_STR, CSXM_STR));
91        }
92        extensions.push(Extensions::N(node));
93    }
94    let fbfx_csxm_nodes = init_fbfx_csxm_fields();
95    for mut node in fbfx_csxm_nodes {
96        if node.get_name() == DE_STR {
97            node.set_content(&format!("{}*", RCJ_STR));
98        }
99        extensions.push(Extensions::N(node));
100    }
101    // 定额下人材机明细Node
102    let rcj_node = init_rcj_fields();
103    extensions.push(Extensions::N(rcj_node));
104    let mut extension = Extension::new();
105    let inc_plugin = Plugin::new(PluginSpec {
106        state_field: Some(Arc::new(IncStateField)),
107        tr: Arc::new(IncStatePlugin),
108    });
109    extension.add_plugin(Arc::new(inc_plugin));
110    extensions.push(Extensions::E(extension));
111    extensions
112}
