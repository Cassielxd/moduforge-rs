14/// 操作符结合性枚举
15/// 定义操作符的结合方向，影响相同优先级操作符的求值顺序
16#[derive(Debug, PartialEq)]
17pub(crate) enum Associativity {
18    Left,  // 左结合：a + b + c = (a + b) + c
19    Right, // 右结合：a ^ b ^ c = a ^ (b ^ c)
20}

30/// 创建HashMap的宏
31/// 提供简洁的语法来初始化HashMap
32macro_rules! hashmap {
33    (@single $($x:tt)*) => (());
34    (@count $($rest:expr),*) => (<[()]>::len(&[$(hashmap!(@single $rest)),*]));
35
36    ($($key:expr => $value:expr,)+) => { hashmap!($($key => $value),+) };
37    ($($key:expr => $value:expr),*) => {
38        {
39            let _cap = hashmap!(@count $($key),*);
40            let mut _map = ::std::collections::HashMap::default();
41            _map.reserve(_cap);
42            $(
43                let _ = _map.insert($key, $value);
44            )*
45            _map
46        }
47    };
48}

50/// 二元操作符优先级和结合性映射表
51/// 定义了所有二元操作符的优先级（数值越大优先级越高）和结合性
52///
53/// 优先级顺序（从低到高）：
54/// - 10: 逻辑或 (or)
55/// - 15: 逻辑与 (and)  
56/// - 20: 比较操作符 (==, !=, <, >, <=, >=, in, not in)
57/// - 30: 加减法 (+, -)
58/// - 60: 乘除取模 (*, /, %)
59/// - 70: 幂运算 (^) - 右结合
60/// - 80: 空值合并 (??)
61pub(crate) static BINARY_OPERATORS: Lazy<
62    HashMap<Operator, ParserOperator, NoHasher>,
63> = Lazy::new(|| {
64    hashmap! {
65        // 逻辑操作符（最低优先级）
66        Operator::Logical(LogicalOperator::Or) => ParserOperator { precedence: 10, associativity: Left },
67        Operator::Logical(LogicalOperator::And) => ParserOperator { precedence: 15, associativity: Left },
68
69        // 比较操作符
70        Operator::Comparison(ComparisonOperator::Equal) => ParserOperator { precedence: 20, associativity: Left },
71        Operator::Comparison(ComparisonOperator::NotEqual) => ParserOperator { precedence: 20, associativity: Left },
72        Operator::Comparison(ComparisonOperator::LessThan) => ParserOperator { precedence: 20, associativity: Left },
73        Operator::Comparison(ComparisonOperator::GreaterThan) => ParserOperator { precedence: 20, associativity: Left },
74        Operator::Comparison(ComparisonOperator::LessThanOrEqual) => ParserOperator { precedence: 20, associativity: Left },
75        Operator::Comparison(ComparisonOperator::GreaterThanOrEqual) => ParserOperator { precedence: 20, associativity: Left },
76        Operator::Comparison(ComparisonOperator::NotIn) => ParserOperator { precedence: 20, associativity: Left },
77        Operator::Comparison(ComparisonOperator::In) => ParserOperator { precedence: 20, associativity: Left },
78
79        // 算术操作符
80        Operator::Arithmetic(ArithmeticOperator::Add) => ParserOperator { precedence: 30, associativity: Left },
81        Operator::Arithmetic(ArithmeticOperator::Subtract) => ParserOperator { precedence: 30, associativity: Left },
82        Operator::Arithmetic(ArithmeticOperator::Multiply) => ParserOperator { precedence: 60, associativity: Left },
83        Operator::Arithmetic(ArithmeticOperator::Divide) => ParserOperator { precedence: 60, associativity: Left },
84        Operator::Arithmetic(ArithmeticOperator::Modulus) => ParserOperator { precedence: 60, associativity: Left },
85        Operator::Arithmetic(ArithmeticOperator::Power) => ParserOperator { precedence: 70, associativity: Right }, // 右结合
86
87        // 特殊操作符
88        Operator::Logical(LogicalOperator::NullishCoalescing) => ParserOperator { precedence: 80, associativity: Left },
89    }
90});
