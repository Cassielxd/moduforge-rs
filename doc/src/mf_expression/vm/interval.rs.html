17/// 虚拟机区间类型
18///
19/// 表示一个区间范围，支持开区间和闭区间。
20/// 可以包含数字或日期类型的边界。
21#[derive(Debug, Clone)]
22pub(crate) struct VmInterval {
23    /// 左括号类型：决定左边界是否包含在区间内
24    pub left_bracket: Bracket,
25    /// 右括号类型：决定右边界是否包含在区间内
26    pub right_bracket: Bracket,
27    /// 左边界值
28    pub left: VmIntervalData,
29    /// 右边界值
30    pub right: VmIntervalData,
31}

47impl VmInterval {
48    /// 将数字区间转换为数组
49    ///
50    /// 如果区间的两个边界都是数字，则将区间转换为包含所有整数的数组。
51    /// 根据括号类型决定是否包含边界值。
52    ///
53    /// # 返回值
54    /// * `Some(Vec<Variable>)` - 包含区间内所有整数的数组
55    /// * `None` - 如果区间包含非数字类型或转换失败
56    pub fn to_array(&self) -> Option<Vec<Variable>> {
57        let (left, right) = match (&self.left, &self.right) {
58            (VmIntervalData::Number(l), VmIntervalData::Number(r)) => (*l, *r),
59            _ => return None,
60        };
61
62        // 根据左括号类型确定起始值
63        let start = match &self.left_bracket {
64            Bracket::LeftParenthesis => left.to_i64()? + 1, // 开区间：不包含左边界
65            Bracket::LeftSquareBracket => left.to_i64()?, // 闭区间：包含左边界
66            _ => return None,
67        };
68
69        // 根据右括号类型确定结束值
70        let end = match &self.right_bracket {
71            Bracket::RightParenthesis => right.to_i64()? - 1, // 开区间：不包含右边界
72            Bracket::RightSquareBracket => right.to_i64()?, // 闭区间：包含右边界
73            _ => return None,
74        };
75
76        // 生成区间内的所有整数
77        let list = (start..=end)
78            .map(|n| Variable::Number(Decimal::from(n)))
79            .collect::<Vec<_>>();
80
81        Some(list)
82    }
83
84    /// 检查值是否在区间内
85    ///
86    /// 根据区间的括号类型和边界值，判断给定值是否包含在区间内。
87    /// 支持开区间、闭区间以及混合区间的判断。
88    ///
89    /// # 参数
90    /// * `v` - 要检查的值
91    ///
92    /// # 返回值
93    /// * `Ok(true)` - 值在区间内
94    /// * `Ok(false)` - 值不在区间内
95    /// * `Err` - 不支持的括号类型或其他错误
96    pub fn includes(
97        &self,
98        v: VmIntervalData,
99    ) -> anyhow::Result<bool> {
100        let mut is_open = false;
101        let l = &self.left;
102        let r = &self.right;
103
104        // 检查左边界条件
105        let first = match &self.left_bracket {
106            Bracket::LeftParenthesis => l < &v, // 开区间：左边界值必须小于v
107            Bracket::LeftSquareBracket => l <= &v, // 闭区间：左边界值可以等于v
108            Bracket::RightParenthesis => {
109                // 开区间的另一种表示
110                is_open = true;
111                l > &v
112            },
113            Bracket::RightSquareBracket => {
114                // 闭区间的另一种表示
115                is_open = true;
116                l >= &v
117            },
118            _ => return Err(anyhow!("Unsupported bracket")),
119        };
120
121        // 检查右边界条件
122        let second = match &self.right_bracket {
123            Bracket::RightParenthesis => r > &v, // 开区间：右边界值必须大于v
124            Bracket::RightSquareBracket => r >= &v, // 闭区间：右边界值可以等于v
125            Bracket::LeftParenthesis => r < &v,  // 开区间的另一种表示
126            Bracket::LeftSquareBracket => r <= &v, // 闭区间的另一种表示
127            _ => return Err(anyhow!("Unsupported bracket")),
128        };
129
130        // 根据区间类型计算最终结果
131        let open_stmt = is_open && (first || second); // 开区间逻辑
132        let closed_stmt = !is_open && first && second; // 闭区间逻辑
133
134        Ok(open_stmt || closed_stmt)
135    }
136}

138impl Display for VmInterval {
139    /// 格式化区间为字符串表示
140    ///
141    /// 格式：`左括号左值..右值右括号`
142    /// 例如：`[1..10]`, `(0..5)`, `[2..8)`
143    fn fmt(
144        &self,
145        f: &mut Formatter<'_>,
146    ) -> std::fmt::Result {
147        write!(
148            f,
149            "{}{}..{}{}",
150            self.left_bracket, self.left, self.right, self.right_bracket
151        )
152    }
153}

155/// 区间数据类型
156///
157/// 表示区间边界可以包含的数据类型
158#[derive(Debug, Clone, PartialOrd, PartialEq, Ord, Eq)]
159pub(crate) enum VmIntervalData {
160    /// 数字类型边界
161    Number(Decimal),
162    /// 日期类型边界
163    Date(VmDate),
164}
165
166impl Display for VmIntervalData {
167    fn fmt(
168        &self,
169        f: &mut Formatter<'_>,
170    ) -> std::fmt::Result {
171        match self {
172            VmIntervalData::Number(n) => write!(f, "{n}"),
173            VmIntervalData::Date(d) => write!(f, "{d}"),
174        }
175    }
