9/// 字段类型信息
10///
11/// 描述一个字段的详细类型信息。
12/// 遵循数据传输对象（DTO）模式，只包含类型分析结果。
13#[derive(Debug, Clone, PartialEq)]
14pub struct FieldTypeInfo {
15    /// 原始类型名称（完整的类型表示）
16    pub original_type: String,
17    
18    /// 简化的类型名称（用于显示和错误消息）
19    pub simple_name: String,
20    
21    /// 是否为 Option<T> 包装类型
22    pub is_optional: bool,
23    
24    /// 内部类型（如果是 Option<T>，则为 T 的类型信息）
25    pub inner_type: Option<Box<FieldTypeInfo>>,
26    
27    /// 是否为支持的基本类型
28    pub is_supported: bool,
29}

31/// 字段分析结果
32///
33/// 包含字段的完整分析结果，包括名称、类型信息和属性标记。
34/// 遵循接口隔离原则，只包含分析结果相关的信息。
35#[derive(Debug, Clone)]
36pub struct FieldAnalysis {
37    /// 字段名称
38    pub name: String,
39    
40    /// 字段的类型信息
41    pub type_info: FieldTypeInfo,
42    
43    /// 是否带有 #[attr] 标记
44    pub is_marked_as_attr: bool,
45    
46    /// 字段的所有属性标记
47    pub attributes: Vec<String>,
48    
49    /// 原始字段引用（用于错误定位）
50    pub original_field: Field,
51}

53/// 字段分析器
54///
55/// 提供字段类型分析和属性检查的核心功能。
56/// 遵循单一职责原则，专门负责字段相关的分析逻辑。
57pub struct FieldAnalyzer;

59impl FieldAnalyzer {
60    /// 分析字段类型信息
61    ///
62    /// 深度分析字段的类型信息，包括是否为 Option 类型、内部类型等。
63    /// 遵循里氏替换原则，任何 Type 都能正确分析。
64    ///
65    /// # 参数
66    ///
67    /// * `field_type` - 要分析的字段类型
68    ///
69    /// # 返回值
70    ///
71    /// 返回详细的字段类型信息
72    ///
73    /// # 分析内容
74    ///
75    /// - 原始类型名称和简化名称
76    /// - 是否为 Option<T> 类型
77    /// - Option 的内部类型信息（递归分析）
78    /// - 是否为支持的基本类型
79    ///
80    /// # 设计原则体现
81    ///
82    /// - **单一职责**: 只负责类型信息分析
83    /// - **里氏替换**: 任何 Type 都能正确处理
84    /// - **开闭原则**: 可扩展支持新的类型分析规则
85    ///
86    /// # 示例
87    ///
88    /// ```rust
89    /// use syn::parse_quote;
90    /// use crate::parser::field_analyzer::FieldAnalyzer;
91    ///
92    /// let ty: Type = parse_quote! { Option<String> };
93    /// let info = FieldAnalyzer::analyze_field_type(&ty);
94    ///
95    /// assert_eq!(info.simple_name, "Option<String>");
96    /// assert!(info.is_optional);
97    /// assert!(info.inner_type.is_some());
98    /// assert_eq!(info.inner_type.unwrap().simple_name, "String");
99    /// ```
100    pub fn analyze_field_type(field_type: &Type) -> FieldTypeInfo {
101        let original_type = quote::quote! { #field_type }.to_string();
102        let simple_name = utils::extract_type_name(field_type);
103        
104        // 检查是否为 Option 类型
105        if utils::is_option_type(field_type) {
106            // 分析 Option 的内部类型
107            if let Some(inner_type) = utils::extract_option_inner_type(field_type) {
108                let inner_info = Self::analyze_field_type(inner_type);
109                
110                FieldTypeInfo {
111                    original_type,
112                    simple_name,
113                    is_optional: true,
114                    inner_type: Some(Box::new(inner_info.clone())),
115                    is_supported: inner_info.is_supported, // Option<T> 的支持性取决于 T
116                }
117            } else {
118                // 无法解析内部类型的 Option
119                FieldTypeInfo {
120                    original_type,
121                    simple_name,
122                    is_optional: true,
123                    inner_type: None,
124                    is_supported: false,
125                }
126            }
127        } else {
128            // 普通类型（非 Option）
129            let is_supported = utils::is_supported_basic_type(field_type);
130            
131            FieldTypeInfo {
132                original_type,
133                simple_name,
134                is_optional: false,
135                inner_type: None,
136                is_supported,
137            }
138        }
139    }

141    /// 分析单个字段
142    ///
143    /// 对字段进行完整分析，包括类型信息和属性标记。
144    /// 遵循单一职责原则，专门负责单个字段的全面分析。
145    ///
146    /// # 参数
147    ///
148    /// * `field` - 要分析的字段
149    ///
150    /// # 返回值
151    ///
152    /// 成功时返回字段分析结果，失败时返回分析错误
153    ///
154    /// # 分析内容
155    ///
156    /// - 字段名称提取和验证
157    /// - 字段类型的详细分析
158    /// - 属性标记的识别和提取
159    /// - 支持性验证
160    ///
161    /// # 设计原则体现
162    ///
163    /// - **单一职责**: 只负责单个字段的分析
164    /// - **接口隔离**: 提供完整但精简的分析结果
165    ///
166    /// # 示例
167    ///
168    /// ```rust
169    /// use syn::parse_quote;
170    /// use crate::parser::field_analyzer::FieldAnalyzer;
171    ///
172    /// let field: Field = parse_quote! {
173    ///     #[attr]
174    ///     name: String
175    /// };
176    ///
177    /// let analysis = FieldAnalyzer::analyze_field(&field)?;
178    /// assert_eq!(analysis.name, "name");
179    /// assert!(analysis.is_marked_as_attr);
180    /// assert!(analysis.type_info.is_supported);
181    /// ```
182    pub fn analyze_field(field: &Field) -> MacroResult<FieldAnalysis> {
183        // 提取字段名称
184        let field_name = field.ident.as_ref()
185            .ok_or_else(|| MacroError::parse_error(
186                "字段缺少名称（不支持匿名字段）",
187                field,
188            ))?
189            .to_string();
190        
191        // 分析字段类型
192        let type_info = Self::analyze_field_type(&field.ty);
193        
194        // 分析字段属性
195        let (is_marked_as_attr, attributes) = Self::analyze_field_attributes(field)?;
196        
197        Ok(FieldAnalysis {
198            name: field_name,
199            type_info,
200            is_marked_as_attr,
201            attributes,
202            original_field: field.clone(),
203        })
204    }

206    /// 分析多个字段
207    ///
208    /// 批量分析多个字段，并返回所有字段的分析结果。
209    /// 遵循开闭原则，可以扩展支持不同的字段过滤和分析策略。
210    ///
211    /// # 参数
212    ///
213    /// * `fields` - 要分析的字段列表
214    ///
215    /// # 返回值
216    ///
217    /// 成功时返回所有字段的分析结果，失败时返回第一个遇到的错误
218    ///
219    /// # 分析策略
220    ///
221    /// - 逐个分析每个字段
222    /// - 收集所有分析结果
223    /// - 遇到错误时立即停止并返回错误
224    ///
225    /// # 设计原则体现
226    ///
227    /// - **单一职责**: 专门负责批量字段分析
228    /// - **开闭原则**: 可扩展不同的批处理策略
229    pub fn analyze_fields(fields: &[Field]) -> MacroResult<Vec<FieldAnalysis>> {
230        let mut analyses = Vec::with_capacity(fields.len());
231        
232        for field in fields {
233            let analysis = Self::analyze_field(field)?;
234            analyses.push(analysis);
235        }
236        
237        Ok(analyses)
238    }

241    ///
242    /// 从字段分析结果中筛选出带有 #[attr] 标记的字段。
243    /// 遵循接口隔离原则，提供专门的过滤功能。
244    ///
245    /// # 参数
246    ///
247    /// * `analyses` - 字段分析结果列表
248    ///
249    /// # 返回值
250    ///
251    /// 返回所有带有 #[attr] 标记的字段分析结果
252    ///
253    /// # 过滤规则
254    ///
255    /// - 只保留 `is_marked_as_attr` 为 true 的字段
256    /// - 保持原有的顺序
257    ///
258    /// # 设计原则体现
259    ///
260    /// - **接口隔离**: 提供专门的过滤接口
261    /// - **单一职责**: 只负责属性字段的筛选
262    pub fn filter_attr_fields(analyses: &[FieldAnalysis]) -> Vec<&FieldAnalysis> {
263        analyses.iter()
264            .filter(|analysis| analysis.is_marked_as_attr)
265            .collect()
266    }

268    /// 验证字段类型的支持性
269    ///
270    /// 检查字段类型是否被宏系统支持。
271    /// 遵循单一职责原则，专门负责类型支持性验证。
272    ///
273    /// # 参数
274    ///
275    /// * `analysis` - 字段分析结果
276    ///
277    /// # 返回值
278    ///
279    /// 如果类型受支持则返回 Ok(())，否则返回验证错误
280    ///
281    /// # 验证规则
282    ///
283    /// - 基本类型必须在支持列表中
284    /// - Option<T> 类型要求 T 是支持的基本类型
285    /// - 复合类型暂不支持
286    ///
287    /// # 设计原则体现
288    ///
289    /// - **单一职责**: 只负责类型支持性验证
290    /// - **开闭原则**: 可扩展支持新的类型验证规则
291    pub fn validate_field_type_support(analysis: &FieldAnalysis) -> MacroResult<()> {
292        if !analysis.type_info.is_supported {
293            return Err(MacroError::unsupported_field_type(
294                &analysis.name,
295                &analysis.type_info.simple_name,
296                &analysis.original_field,
297            ));
298        }
299        
300        // 对于 Option 类型，还需要验证内部类型
301        if analysis.type_info.is_optional {
302            if let Some(inner_type) = &analysis.type_info.inner_type {
303                if !inner_type.is_supported {
304                    return Err(MacroError::unsupported_field_type(
305                        &analysis.name,
306                        &inner_type.simple_name,
307                        &analysis.original_field,
308                    ));
309                }
310            }
311        }
312        
313        Ok(())
314    }

317    ///
318    /// 对多个字段进行类型支持性验证。
319    /// 遵循开闭原则，可以扩展不同的批量验证策略。
320    ///
321    /// # 参数
322    ///
323    /// * `analyses` - 字段分析结果列表
324    ///
325    /// # 返回值
326    ///
327    /// 如果所有字段类型都受支持则返回 Ok(())，否则返回第一个验证错误
328    ///
329    /// # 验证策略
330    ///
331    /// - 逐个验证每个字段的类型支持性
332    /// - 遇到不支持的类型时立即返回错误
333    /// - 提供详细的错误信息
334    ///

336    ///
337    /// - **单一职责**: 专门负责批量类型验证
338    /// - **里氏替换**: 可以替换单个字段验证使用

346    /// 分析字段的属性标记
347    ///
348    /// 提取和分析字段上的所有属性标记。
349    /// 遵循单一职责原则，专门处理字段属性的提取。
350    ///
351    /// # 参数
352    ///
353    /// * `field` - 要分析的字段
354    ///
355    /// # 返回值
356    ///
357    /// 返回元组：(是否带有 #[attr] 标记, 所有属性名称列表)
358    ///
359    /// # 分析内容
360    ///
361    /// - 检查是否带有 #[attr] 标记
362    /// - 提取所有属性的名称
363    /// - 过滤无效的属性
364    ///
365    /// # 设计原则体现
366    ///
367    /// - **单一职责**: 只负责属性标记的分析
368    /// - **接口隔离**: 提供简洁的属性分析接口
369    fn analyze_field_attributes(field: &Field) -> MacroResult<(bool, Vec<String>)> {
370        let mut is_marked_as_attr = false;
371        let mut attributes = Vec::new();
372        
373        for attr in &field.attrs {
374            if let Some(ident) = attr.path().get_ident() {
375                let attr_name = ident.to_string();
376                attributes.push(attr_name.clone());
377                
378                // 检查是否为 #[attr] 标记
379                if attr_name == "attr" {
380                    is_marked_as_attr = true;
381                    
382                    // 验证 #[attr] 属性的格式
383                    Self::validate_attr_attribute(attr)?;
384                }
385            }
386        }
387        
388        Ok((is_marked_as_attr, attributes))
389    }
390    
391    /// 验证 #[attr] 属性的格式
392    ///
393    /// 确保 #[attr] 属性使用正确的格式。
394    /// 遵循单一职责原则，专门验证属性格式。
395    ///
396    /// # 参数
397    ///
398    /// * `attr` - 要验证的属性
399    ///
400    /// # 返回值
401    ///
402    /// 如果格式正确则返回 Ok(())，否则返回格式错误
403    ///
404    /// # 验证规则
405    ///
406    /// - #[attr] 应该是简单标记，不带参数
407    /// - 不支持 #[attr = "value"] 或 #[attr(param)] 格式
408    ///
409    /// # 设计原则体现
410    ///
411    /// - **单一职责**: 只负责 #[attr] 属性格式验证
412    /// - **接口隔离**: 提供专门的格式验证接口
413    fn validate_attr_attribute(attr: &syn::Attribute) -> MacroResult<()> {
414        match &attr.meta {
415            syn::Meta::Path(_) => {
416                // #[attr] 格式，正确
417                Ok(())
418            }
419            syn::Meta::List(_) => {
420                // #[attr(...)] 格式，暂不支持
421                Err(MacroError::parse_error(
422                    "#[attr] 不支持参数，请使用简单的 #[attr] 标记",
423                    attr,
424                ))
425            }
426            syn::Meta::NameValue(_) => {
427                // #[attr = "..."] 格式，暂不支持
428                Err(MacroError::parse_error(
429                    "#[attr] 不支持值赋值，请使用简单的 #[attr] 标记",
430                    attr,
431                ))
432            }
433        }
434    }

437impl FieldTypeInfo {
438    /// 获取用于代码生成的类型名称
439    ///
440    /// 返回适合在生成代码中使用的类型名称。
441    /// 遵循单一职责原则，专门提供代码生成所需的类型名称。
442    ///
443    /// # 返回值
444    ///
445    /// 返回用于代码生成的类型名称字符串
446    ///
447    /// # 命名规则
448    ///
449    /// - 基本类型返回简化名称
450    /// - Option<T> 类型返回完整的 Option<T> 表示
451    /// - 复杂类型返回原始类型表示
452    ///
453    /// # 设计原则体现
454    ///
455    /// - **单一职责**: 只负责提供代码生成用的类型名称
456    /// - **接口隔离**: 提供专门的类型名称获取接口
457    pub fn codegen_type_name(&self) -> &str {
458        &self.simple_name
459    }
460    
461    /// 获取基础类型名称
462    ///
463    /// 返回去除 Option 包装后的基础类型名称。
464    /// 用于类型检查和代码生成逻辑。
465    ///
466    /// # 返回值
467    ///
468    /// 返回基础类型名称，如果是 Option<T> 则返回 T 的名称
469    ///
470    /// # 示例
471    ///
472    /// ```rust
473    /// // 对于 String 类型
474    /// assert_eq!(type_info.base_type_name(), "String");
475    ///
476    /// // 对于 Option<String> 类型
477    /// assert_eq!(type_info.base_type_name(), "String");
478    /// ```
479    ///
480    /// # 设计原则体现
481    ///
482    /// - **单一职责**: 只负责提取基础类型名称
483    /// - **里氏替换**: 任何类型信息都能正确处理
484    pub fn base_type_name(&self) -> &str {
485        if self.is_optional {
486            if let Some(inner_type) = &self.inner_type {
487                inner_type.base_type_name()
488            } else {
489                &self.simple_name
490            }
491        } else {
492            &self.simple_name
493        }
494    }
495    
496    /// 检查是否为字符串类型
497    ///
498    /// 判断类型是否为字符串相关类型（String, &str, str）。
499    /// 用于特殊的字符串处理逻辑。
500    ///
501    /// # 返回值
502    ///
503    /// 如果是字符串类型则返回 true，否则返回 false
504    ///
505    /// # 字符串类型识别
506    ///
507    /// - String
508    /// - &str
509    /// - str
510    /// - Option<String>, Option<&str>, Option<str>
511    ///
512    /// # 设计原则体现
513    ///
514    /// - **单一职责**: 只负责字符串类型识别
515    /// - **开闭原则**: 可扩展支持新的字符串类型
516    pub fn is_string_type(&self) -> bool {
517        let base_name = self.base_type_name();
518        matches!(base_name, "String" | "str" | "&str")
519    }
520    
521    /// 检查是否为数值类型
522    ///
523    /// 判断类型是否为数值相关类型。
524    /// 用于数值处理的特殊逻辑。
525    ///
526    /// # 返回值
527    ///
528    /// 如果是数值类型则返回 true，否则返回 false
529    ///
530    /// # 数值类型识别
531    ///
532    /// - 整数类型: i8, i16, i32, i64, i128, isize, u8, u16, u32, u64, u128, usize
533    /// - 浮点类型: f32, f64
534    /// - 对应的 Option 包装版本
535    ///
536    /// # 设计原则体现
537    ///
538    /// - **单一职责**: 只负责数值类型识别
539    /// - **开闭原则**: 可扩展支持新的数值类型
540    pub fn is_numeric_type(&self) -> bool {
541        let base_name = self.base_type_name();
542        matches!(base_name, 
543            "i8" | "i16" | "i32" | "i64" | "i128" | "isize" |
544            "u8" | "u16" | "u32" | "u64" | "u128" | "usize" |
545            "f32" | "f64"
546        )
547    }

550impl FieldAnalysis {
551    /// 检查字段是否可用作属性
552    ///
553    /// 验证字段是否满足作为属性使用的所有条件。
554    /// 结合了属性标记和类型支持性的检查。
555    ///
556    /// # 返回值
557    ///
558    /// 如果字段可用作属性则返回 Ok(())，否则返回验证错误
559    ///
560    /// # 验证条件
561    ///
562    /// - 必须带有 #[attr] 标记
563    /// - 字段类型必须受支持
564    /// - 字段名称必须是有效标识符
565    ///
566    /// # 设计原则体现
567    ///
568    /// - **单一职责**: 只负责属性字段的可用性验证
569    /// - **接口隔离**: 提供综合的验证接口
570    pub fn validate_as_attribute(&self) -> MacroResult<()> {
571        // 检查是否带有 #[attr] 标记
572        if !self.is_marked_as_attr {
573            return Err(MacroError::validation_error(
574                &format!("字段 '{}' 没有 #[attr] 标记", self.name),
575                &self.original_field,
576            ));
577        }
578        
579        // 检查字段名称是否为有效标识符
580        if !utils::is_valid_identifier(&self.name) {
581            return Err(MacroError::validation_error(
582                &format!("字段名称 '{}' 不是有效的标识符", self.name),
583                &self.original_field,
584            ));
585        }
586        
587        // 检查类型支持性
588        FieldAnalyzer::validate_field_type_support(self)?;
589        
590        Ok(())
591    }

593    /// 生成字段的显示信息
594    ///
595    /// 创建用于错误消息和调试的字段描述信息。
596    /// 遵循单一职责原则，专门提供字段信息的格式化。
597    ///
598    /// # 返回值
599    ///
600    /// 返回包含字段详细信息的字符串
601    ///
602    /// # 信息内容
603    ///
604    /// - 字段名称
605    /// - 字段类型
606    /// - 是否为可选类型
607    /// - 是否带有属性标记
608    /// - 类型支持状态
609    ///
610    /// # 设计原则体现
611    ///
612    /// - **单一职责**: 只负责字段信息的格式化
613    /// - **接口隔离**: 提供统一的信息显示接口
614    pub fn display_info(&self) -> String {
615        format!(
616            "字段 '{}': {} ({}{}{})",
617            self.name,
618            self.type_info.simple_name,
619            if self.type_info.is_optional { "可选, " } else { "必需, " },
620            if self.is_marked_as_attr { "带属性标记, " } else { "无属性标记, " },
621            if self.type_info.is_supported { "支持" } else { "不支持" }
622        )
623    }
624}
