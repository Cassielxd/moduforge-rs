12/// 字符串类型转换器
13///
14/// 专门处理字符串相关类型的转换，包括 String、&str、str 和它们的 Option 版本。
15/// 遵循单一职责原则，只负责字符串类型的转换逻辑。
16///
17/// # 支持的类型
18///
19/// - String
20/// - &str (引用字符串)
21/// - str (字符串切片)
22/// - Option<String>
23/// - Option<&str>
24/// - Option<str>
25///
26/// # 设计原则体现
27///
28/// - **单一职责**: 只负责字符串类型转换
29/// - **里氏替换**: 实现了 TypeConverter，可替换使用
30/// - **接口隔离**: 提供专门的字符串转换接口
31#[derive(Debug, Clone)]
32pub struct StringConverter;
33
34impl StringConverter {
35    /// 创建新的字符串转换器实例
36    ///
37    /// # 返回值
38    ///
39    /// 返回配置好的字符串转换器
40    pub fn new() -> Self {
41        Self
42    }
43
44    /// 检查是否为字符串类型
45    ///
46    /// 判断给定类型是否为字符串相关类型。
47    /// 遵循单一职责原则，专门负责字符串类型识别。
48    ///
49    /// # 参数
50    ///
51    /// * `field_type` - 要检查的字段类型
52    ///
53    /// # 返回值
54    ///
55    /// 如果是字符串类型则返回 true，否则返回 false
56    ///
57    /// # 识别逻辑
58    ///
59    /// - 对于普通类型，检查是否为 String、&str、str
60    /// - 对于 Option<T>，检查内部类型 T 是否为字符串类型
61    ///
62    /// # 设计原则体现
63    ///
64    /// - **单一职责**: 只负责字符串类型识别
65    /// - **里氏替换**: 任何字符串类型都能正确识别
66    fn is_string_type(&self, field_type: &Type) -> bool {
67        if utils::is_option_type(field_type) {
68            // Option<T> 类型，检查内部类型
69            if let Some(inner_type) = utils::extract_option_inner_type(field_type) {
70                self.is_basic_string_type(inner_type)
71            } else {
72                false
73            }
74        } else {
75            // 普通类型
76            self.is_basic_string_type(field_type)
77        }
78    }
79
80    /// 检查是否为基本字符串类型
81    ///
82    /// 判断类型是否为基本的字符串类型（不包括 Option 包装）。
83    /// 遵循单一职责原则，专门负责基本字符串类型检查。
84    ///
85    /// # 参数
86    ///
87    /// * `field_type` - 要检查的字段类型
88    ///
89    /// # 返回值
90    ///
91    /// 如果是基本字符串类型则返回 true，否则返回 false
92    fn is_basic_string_type(&self, field_type: &Type) -> bool {
93        let type_name = utils::extract_type_name(field_type);
94        matches!(type_name.as_str(), "String" | "str" | "&str")
95    }
96}
97
98impl Default for StringConverter {
99    fn default() -> Self {
100        Self::new()
101    }
102}

104impl TypeConverter for StringConverter {
105    fn convert_field_to_json_value(&self, field: &Field) -> MacroResult<TokenStream2> {
106        if !self.supports_type(&field.ty) {
107            let type_name = utils::extract_type_name(&field.ty);
108            let field_name = field.ident.as_ref()
109                .map(|i| i.to_string())
110                .unwrap_or_else(|| "匿名字段".to_string());
111
112            return Err(MacroError::unsupported_field_type(
113                &field_name,
114                &type_name,
115                field,
116            ));
117        }
118
119        let field_name = field.ident.as_ref()
120            .ok_or_else(|| MacroError::parse_error(
121                "字段缺少名称（不支持匿名字段）",
122                field,
123            ))?;
124
125        // 生成字符串转换代码
126        let conversion_code = utils::generate_field_conversion(field_name, &field.ty);
127        Ok(conversion_code)
128    }
129
130    fn supports_type(&self, field_type: &Type) -> bool {
131        self.is_string_type(field_type)
132    }
133
134    fn priority(&self) -> i32 {
135        90 // 比通用转换器高，但比数值转换器低
136    }
137
138    fn name(&self) -> &'static str {
139        "StringConverter"
140    }
141}

431/// 获取所有内置转换器
432///
433/// 返回一个包含所有内置转换器的向量。
434/// 遵循工厂模式，统一创建和管理内置转换器实例。
435///
436/// # 返回值
437///
438/// 返回包含所有内置转换器的 Box<dyn TypeConverter> 向量
439///
440/// # 转换器顺序
441///
442/// 转换器按优先级排序（高到低）：
443/// 1. NumericConverter (95) - 数值类型转换器
444/// 2. StringConverter (90) - 字符串类型转换器  
445/// 3. BooleanConverter (85) - 布尔类型转换器
446/// 4. SpecialTypeConverter (50) - 特殊类型转换器
447///
448/// # 设计原则体现
449///
450/// - **工厂模式**: 统一创建转换器实例
451/// - **开闭原则**: 可以扩展新的内置转换器而不修改调用方
452/// - **依赖注入**: 返回接口类型而非具体类型
453///
454/// # 示例
455///
456/// ```rust
457/// use crate::converter::builtin_converters::get_all_builtin_converters;
458///
459/// let converters = get_all_builtin_converters();
460/// assert_eq!(converters.len(), 4);
461///
462/// // 验证转换器优先级顺序
463/// assert!(converters[0].priority() >= converters[1].priority());
464/// ```
465pub fn get_all_builtin_converters() -> Vec<Box<dyn TypeConverter>> {
466    let mut converters: Vec<Box<dyn TypeConverter>> = vec![
467        Box::new(NumericConverter::new()),
468        Box::new(StringConverter::new()),
469        Box::new(BooleanConverter::new()),
470        Box::new(SpecialTypeConverter::new()),
471    ];
472
473    // 按优先级降序排序
474    converters.sort_by(|a, b| b.priority().cmp(&a.priority()));
475
476    converters
477}
