1//! 类型转换器模块
2//!
3//! 提供完整的类型转换系统，支持 Rust 基本类型到 `serde_json::Value` 的转换。
4//! 严格遵循 SOLID 设计原则，确保系统的可扩展性和可维护性。
5//!
6//! # 模块组成
7//!
8//! - `type_converter`: 类型转换器核心接口和基础实现
9//! - `builtin_converters`: 内置转换器实现（字符串、数值、布尔等类型）
10//! - `converter_registry`: 转换器注册表，支持全局转换器管理
11//!
12//! # 设计原则体现
13//!
14//! - **单一职责原则 (SRP)**: 每个转换器都专注于特定类型的转换
15//! - **开闭原则 (OCP)**: 通过实现 TypeConverter trait 扩展新的转换器
16//! - **里氏替换原则 (LSP)**: 所有转换器都可以无缝替换使用
17//! - **接口隔离原则 (ISP)**: 提供专门的转换接口，不强制依赖不需要的功能
18//! - **依赖倒置原则 (DIP)**: 依赖于抽象接口而非具体实现
19//!
20//! # 使用方式
21//!
22//! ## 基本使用
23//!
24//! ```rust
25//! use syn::parse_quote;
26//! use crate::converter::converter_registry::GlobalConverterRegistry;
27//!
28//! // 转换字段
29//! let field: syn::Field = parse_quote! {
30//!     name: String
31//! };
32//!
33//! let conversion_code = GlobalConverterRegistry::convert_field(&field)?;
34//! ```
35//!
36//! ## 注册自定义转换器
37//!
38//! ```rust
39//! use crate::converter::{
40//!     type_converter::TypeConverter,
41//!     converter_registry::GlobalConverterRegistry
42//! };
43//!
44//! // 实现自定义转换器
45//! struct CustomConverter;
46//!
47//! impl TypeConverter for CustomConverter {
48//!     fn convert_field_to_json_value(&self, field: &syn::Field) -> MacroResult<TokenStream2> {
49//!         // 自定义转换逻辑
50//!         todo!()
51//!     }
52//!
53//!     fn supports_type(&self, field_type: &syn::Type) -> bool {
54//!         // 支持性检查
55//!         false
56//!     }
57//!
58//!     fn priority(&self) -> i32 {
59//!         200 // 高优先级
60//!     }
61//! }
62//!
63//! // 注册到全局注册表
64//! GlobalConverterRegistry::register(Box::new(CustomConverter))?;
65//! ```
66//!
67//! # 支持的类型
68//!
69//! ## 基本类型
70//!
71//! - **字符串**: String, &str, str
72//! - **整数**: i8, i16, i32, i64, i128, isize, u8, u16, u32, u64, u128, usize
73//! - **浮点数**: f32, f64
74//! - **布尔**: bool
75//!
76//! ## 可选类型
77//!
78//! - **Option<T>**: 其中 T 是上述任意支持的基本类型
79//!
80//! # 转换器优先级
81//!
82//! 转换器按优先级排序，数值越大优先级越高：
83//!
84//! - **BuiltinTypeConverter**: 100 - 通用基本类型转换器
85//! - **NumericConverter**: 95 - 专门的数值类型转换器
86//! - **StringConverter**: 90 - 专门的字符串类型转换器
87//! - **BooleanConverter**: 85 - 专门的布尔类型转换器
88//! - **SpecialTypeConverter**: 50 - 特殊类型转换器（预留）
89//!
90//! # 错误处理
91//!
92//! 转换过程中可能出现的错误：
93//!
94//! - **UnsupportedFieldType**: 字段类型不受支持
95//! - **ParseError**: 字段解析错误（如缺少名称）
96//! - **GenerationError**: 代码生成过程中的错误
97//!
98//! # 线程安全
99//!
100//! 全局转换器注册表是线程安全的：
101//! - 读操作（转换）可以并发执行
102//! - 写操作（注册）会独占锁
103//! - 所有操作都有适当的错误处理
104//!
105//! # 性能考虑
106//!
107//! - 转换器按优先级预排序，查找效率高
108//! - 使用 `once_cell` 实现全局单例，初始化开销低
109//! - 代码生成采用零开销抽象，运行时无额外成本

147    /// 便利函数：转换字段
148    ///
149    /// 使用全局注册表转换字段的快捷方式。
150    /// 等同于 `GlobalConverterRegistry::convert_field(field)`。
151    ///
152    /// # 参数
153    ///
154    /// * `field` - 要转换的字段
155    ///
156    /// # 返回值
157    ///
158    /// 成功时返回转换代码，失败时返回错误
159    ///
160    /// # 示例
161    ///
162    /// ```rust
163    /// use syn::parse_quote;
164    /// use crate::converter::utils::convert_field;
165    ///
166    /// let field: syn::Field = parse_quote! {
167    ///     name: String
168    /// };
169    ///
170    /// let code = convert_field(&field)?;
171    /// ```
172    ///
173    /// # 设计原则体现
174    ///
175    /// - **门面模式**: 提供简化的接口
176    /// - **单一职责**: 只负责字段转换
177    pub fn convert_field(field: &Field) -> MacroResult<TokenStream2> {
178        GlobalConverterRegistry::convert_field(field)
179    }

181    /// 便利函数：检查类型支持性
182    ///
183    /// 检查全局注册表是否支持指定类型的快捷方式。
184    ///
185    /// # 参数
186    ///
187    /// * `field_type` - 要检查的字段类型
188    ///
189    /// # 返回值
190    ///
191    /// 如果类型受支持则返回 true，否则返回 false
192    /// 如果访问全局注册表失败则返回 false
193    ///
194    /// # 示例
195    ///
196    /// ```rust
197    /// use syn::parse_quote;
198    /// use crate::converter::utils::is_type_supported;
199    ///
200    /// let string_type: syn::Type = parse_quote! { String };
201    /// assert!(is_type_supported(&string_type));
202    ///
203    /// let unsupported_type: syn::Type = parse_quote! { Vec<String> };
204    /// assert!(!is_type_supported(&unsupported_type));
205    /// ```
206    ///
207    /// # 设计原则体现
208    ///
209    /// - **门面模式**: 提供简化的检查接口
210    /// - **单一职责**: 只负责支持性检查
211    pub fn is_type_supported(field_type: &Type) -> bool {
212        GlobalConverterRegistry::supports_type(field_type).unwrap_or(false)
213    }

215    /// 便利函数：获取支持类型列表
216    ///
217    /// 返回当前支持的所有基本类型的列表。
218    /// 主要用于文档生成和错误提示。
219    ///
220    /// # 返回值
221    ///
222    /// 返回支持的类型名称列表
223    ///
224    /// # 设计原则体现
225    ///
226    /// - **信息隐藏**: 提供类型信息而不暴露内部实现
227    /// - **单一职责**: 只负责提供类型信息
228    pub fn get_supported_types() -> Vec<&'static str> {
229        vec![
230            // 字符串类型
231            "String", "&str", "str",
232            // 有符号整数类型
233            "i8", "i16", "i32", "i64", "i128", "isize",
234            // 无符号整数类型
235            "u8", "u16", "u32", "u64", "u128", "usize",
236            // 浮点数类型
237            "f32", "f64",
238            // 布尔类型
239            "bool",
240        ]
241    }

243    /// 便利函数：生成支持类型的错误提示
244    ///
245    /// 生成友好的错误提示信息，列出所有支持的类型。
246    /// 用于改善用户体验的错误消息。
247    ///
248    /// # 返回值
249    ///
250    /// 返回包含支持类型列表的错误提示字符串
251    ///
252    /// # 示例
253    ///
254    /// ```rust
255    /// use crate::converter::utils::generate_supported_types_hint;
256    ///
257    /// let hint = generate_supported_types_hint();
258    /// assert!(hint.contains("String"));
259    /// assert!(hint.contains("i32"));
260    /// assert!(hint.contains("bool"));
261    /// ```
262    ///
263    /// # 设计原则体现
264    ///
265    /// - **用户体验**: 提供友好的错误提示
266    /// - **信息封装**: 统一错误消息格式
267    pub fn generate_supported_types_hint() -> String {
268        let basic_types = get_supported_types();
269        let basic_list = basic_types.join(", ");
270
271        format!(
272            "支持的基本类型: {}\n\
273            支持的可选类型: Option<T> (其中 T 是上述任意基本类型)\n\
274            \n\
275            示例：\n\
276            - String, Option<String>\n\
277            - i32, Option<i32>\n\
278            - bool, Option<bool>\n\
279            \n\
280            如需支持其他类型，请实现自定义 TypeConverter 并注册到全局注册表。",
281            basic_list
282        )
283    }

397            // 验证错误类型
398            if let Err(error) = result {

400                    crate::common::MacroError::UnsupportedFieldType { .. } => {
401                        // 正确的错误类型
402                    }
403                    _ => panic!("期望 UnsupportedFieldType 错误"),

405            }
