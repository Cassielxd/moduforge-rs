16/// 全局转换器注册表
17///
18/// 使用 `once_cell` 实现全局单例，确保转换器注册表在整个程序生命周期中只有一个实例。
19/// 遵循单例模式，提供全局统一的转换器管理。
20///
21/// # 线程安全
22///
23/// 使用 `RwLock` 确保多线程环境下的安全访问：
24/// - 读取操作可以并发进行
25/// - 写入操作（注册新转换器）会独占锁
26///
27/// # 设计原则体现
28///
29/// - **单例模式**: 确保全局只有一个注册表实例
30/// - **线程安全**: 支持多线程环境下的安全使用
31/// - **开闭原则**: 支持运行时添加新转换器
32static GLOBAL_REGISTRY: Lazy<Arc<RwLock<ConverterRegistryImpl>>> = Lazy::new(|| {
33    let registry = ConverterRegistryImpl::new();
34    Arc::new(RwLock::new(registry))
35});

37/// 转换器注册表接口
38///
39/// 定义转换器注册表的核心接口，遵循接口隔离原则。
40/// 只包含注册表相关的必要方法，不涉及具体的转换逻辑。
41///
42/// # 设计原则体现
43///
44/// - **接口隔离**: 只定义注册表相关的必要方法
45/// - **单一职责**: 专门负责转换器的管理和选择
46pub trait ConverterRegistry {
47    /// 注册新的转换器
48    ///
49    /// 将转换器添加到注册表中，按优先级进行排序。
50    /// 高优先级的转换器会优先被选择使用。
51    ///
52    /// # 参数
53    ///
54    /// * `converter` - 要注册的转换器
55    ///
56    /// # 设计原则体现
57    ///
58    /// - **开闭原则**: 支持扩展新的转换器而不修改现有代码
59    /// - **依赖注入**: 接受任何实现了 TypeConverter 的类型
60    fn register_converter(&mut self, converter: Box<dyn TypeConverter>);
61
62    /// 查找支持指定类型的转换器
63    ///
64    /// 根据字段类型查找合适的转换器。
65    /// 返回第一个支持该类型的转换器，按优先级排序。
66    ///
67    /// # 参数
68    ///
69    /// * `field_type` - 要转换的字段类型
70    ///
71    /// # 返回值
72    ///
73    /// 如果找到支持的转换器则返回 Some，否则返回 None
74    ///
75    /// # 设计原则体现
76    ///
77    /// - **策略模式**: 根据类型选择合适的转换策略
78    /// - **单一职责**: 只负责转换器查找，不执行转换
79    fn find_converter_for_type(&self, field_type: &Type) -> Option<&dyn TypeConverter>;
80
81    /// 执行字段转换
82    ///
83    /// 为字段查找合适的转换器并执行转换。
84    /// 这是注册表的主要功能接口。
85    ///
86    /// # 参数
87    ///
88    /// * `field` - 要转换的字段
89    ///
90    /// # 返回值
91    ///
92    /// 成功时返回转换代码，失败时返回错误
93    ///
94    /// # 错误情况
95    ///
96    /// - 没有找到支持该类型的转换器
97    /// - 转换过程中发生错误
98    ///
99    /// # 设计原则体现
100    ///
101    /// - **门面模式**: 提供统一的转换接口
102    /// - **策略模式**: 内部选择合适的转换策略
103    fn convert_field(&self, field: &Field) -> MacroResult<TokenStream2>;
104
105    /// 获取所有注册的转换器数量
106    ///
107    /// 返回当前注册表中转换器的总数。
108    /// 主要用于调试和测试。
109    ///
110    /// # 返回值
111    ///
112    /// 返回转换器的数量
113    fn converter_count(&self) -> usize;
114
115    /// 清空所有转换器
116    ///
117    /// 移除注册表中的所有转换器。
118    /// 主要用于测试环境的清理。
119    ///
120    /// # 设计原则体现
121    ///
122    /// - **单一职责**: 只负责清理操作
123    fn clear_converters(&mut self);
124
125    /// 重新加载默认转换器
126    ///
127    /// 清空注册表并重新加载所有内置转换器。
128    /// 用于重置注册表到初始状态。
129    ///
130    /// # 设计原则体现
131    ///
132    /// - **工厂模式**: 重新创建默认转换器集合
133    fn reload_default_converters(&mut self);
134}

136/// 转换器注册表实现
137///
138/// ConverterRegistry 接口的具体实现。
139/// 遵循单一职责原则，专门负责转换器的管理和选择逻辑。
140///
141/// # 内部结构
142///
143/// - `converters`: 存储所有注册的转换器，按优先级降序排列
144///
145/// # 设计原则体现
146///
147/// - **单一职责**: 只负责转换器管理
148/// - **里氏替换**: 实现了 ConverterRegistry 接口
149/// - **开闭原则**: 支持运行时添加新转换器
150pub struct ConverterRegistryImpl {
151    /// 转换器列表，按优先级降序排列
152    ///
153    /// 优先级高的转换器排在前面，查找时会优先选择。
154    /// 使用 Vec 而不是 HashMap 是为了保持优先级顺序。
155    converters: Vec<Box<dyn TypeConverter>>,
156}
157
158impl ConverterRegistryImpl {
159    /// 创建新的转换器注册表
160    ///
161    /// 初始化注册表并加载所有内置转换器。
162    /// 内置转换器按优先级自动排序。
163    ///
164    /// # 返回值
165    ///
166    /// 返回包含所有内置转换器的注册表实例
167    ///
168    /// # 初始化的转换器
169    ///
170    /// - NumericConverter (优先级 95)
171    /// - StringConverter (优先级 90)
172    /// - BooleanConverter (优先级 85)
173    /// - SpecialTypeConverter (优先级 50)
174    ///
175    /// # 设计原则体现
176    ///
177    /// - **工厂模式**: 自动创建和配置内置转换器
178    /// - **单一职责**: 只负责注册表的初始化
179    pub fn new() -> Self {
180        let mut registry = Self {
181            converters: Vec::new(),
182        };
183
184        // 加载所有内置转换器
185        registry.load_builtin_converters();
186
187        registry
188    }
189
190    /// 加载内置转换器
191    ///
192    /// 从 builtin_converters 模块获取所有内置转换器并注册。
193    /// 遵循单一职责原则，专门负责内置转换器的加载。
194    ///
195    /// # 加载过程
196    ///
197    /// 1. 获取所有内置转换器
198    /// 2. 按优先级排序
199    /// 3. 注册到当前注册表
200    ///
201    /// # 设计原则体现
202    ///
203    /// - **单一职责**: 只负责内置转换器加载
204    /// - **开闭原则**: 新增内置转换器无需修改此方法
205    fn load_builtin_converters(&mut self) {
206        let builtin_converters = get_all_builtin_converters();
207        for converter in builtin_converters {
208            self.register_converter_internal(converter);
209        }
210    }
211
212    /// 内部转换器注册方法
213    ///
214    /// 执行实际的转换器注册逻辑，包括优先级排序。
215    /// 遵循单一职责原则，专门负责转换器的插入和排序。
216    ///
217    /// # 参数
218    ///
219    /// * `converter` - 要注册的转换器
220    ///
221    /// # 注册逻辑
222    ///
223    /// 1. 查找合适的插入位置（按优先级降序）
224    /// 2. 插入转换器到正确位置
225    /// 3. 保持列表的优先级顺序
226    ///
227    /// # 设计原则体现
228    ///
229    /// - **单一职责**: 只负责转换器插入和排序
230    /// - **开闭原则**: 支持任意类型的转换器插入
231    fn register_converter_internal(&mut self, converter: Box<dyn TypeConverter>) {
232        let priority = converter.priority();
233
234        // 找到合适的插入位置（保持优先级降序）
235        let insert_pos = self.converters
236            .iter()
237            .position(|c| c.priority() < priority)
238            .unwrap_or(self.converters.len());
239
240        self.converters.insert(insert_pos, converter);
241    }
242}
243
244impl Default for ConverterRegistryImpl {
245    /// 创建默认的转换器注册表
246    ///
247    /// 等同于 `ConverterRegistryImpl::new()`
248    fn default() -> Self {
249        Self::new()
250    }
251}
252
253impl ConverterRegistry for ConverterRegistryImpl {
254    fn register_converter(&mut self, converter: Box<dyn TypeConverter>) {
255        self.register_converter_internal(converter);
256    }
257
258    fn find_converter_for_type(&self, field_type: &Type) -> Option<&dyn TypeConverter> {
259        self.converters
260            .iter()
261            .find(|converter| converter.supports_type(field_type))
262            .map(|boxed| boxed.as_ref())
263    }
264
265    fn convert_field(&self, field: &Field) -> MacroResult<TokenStream2> {
266        // 查找支持该字段类型的转换器
267        if let Some(converter) = self.find_converter_for_type(&field.ty) {
268            converter.convert_field_to_json_value(field)
269        } else {
270            // 没有找到支持的转换器
271            let type_name = crate::common::utils::extract_type_name(&field.ty);
272            let field_name = field.ident.as_ref()
273                .map(|i| i.to_string())
274                .unwrap_or_else(|| "匿名字段".to_string());
275
276            Err(MacroError::unsupported_field_type(
277                &field_name,
278                &type_name,
279                field,
280            ))
281        }
282    }
283
284    fn converter_count(&self) -> usize {
285        self.converters.len()
286    }
287
288    fn clear_converters(&mut self) {
289        self.converters.clear();
290    }
291
292    fn reload_default_converters(&mut self) {
293        self.clear_converters();
294        self.load_builtin_converters();
295    }
296}

298/// 全局转换器注册表管理器
299///
300/// 提供对全局转换器注册表的访问和管理功能。
301/// 遵循门面模式，隐藏全局注册表的复杂性。
302///
303/// # 设计原则体现
304///
305/// - **门面模式**: 提供简化的全局注册表访问接口
306/// - **单一职责**: 专门负责全局注册表的管理
307/// - **线程安全**: 所有操作都是线程安全的
308pub struct GlobalConverterRegistry;
309
310impl GlobalConverterRegistry {
311    /// 注册全局转换器
312    ///
313    /// 将转换器注册到全局注册表中。
314    /// 这是添加自定义转换器的主要入口点。
315    ///
316    /// # 参数
317    ///
318    /// * `converter` - 要注册的转换器
319    ///
320    /// # 返回值
321    ///
322    /// 成功时返回 Ok(())，失败时返回锁定错误
323    ///
324    /// # 线程安全
325    ///
326    /// 此方法是线程安全的，可以在多线程环境中调用。
327    ///
328    /// # 示例
329    ///
330    /// ```rust
331    /// use crate::converter::{converter_registry::GlobalConverterRegistry, type_converter::BuiltinTypeConverter};
332    ///
333    /// let custom_converter = Box::new(BuiltinTypeConverter::new());
334    /// GlobalConverterRegistry::register(custom_converter).unwrap();
335    /// ```
336    ///
337    /// # 设计原则体现
338    ///
339    /// - **门面模式**: 隐藏全局锁的复杂性
340    /// - **开闭原则**: 支持运行时扩展转换器
341    pub fn register(converter: Box<dyn TypeConverter>) -> Result<(), String> {
342        match GLOBAL_REGISTRY.write() {
343            Ok(mut registry) => {
344                registry.register_converter(converter);
345                Ok(())
346            }
347            Err(e) => Err(format!("无法获取注册表写锁: {}", e)),
348        }
349    }
350
351    /// 转换字段
352    ///
353    /// 使用全局注册表转换字段。
354    /// 这是执行字段转换的主要入口点。
355    ///
356    /// # 参数
357    ///
358    /// * `field` - 要转换的字段
359    ///
360    /// # 返回值
361    ///
362    /// 成功时返回转换代码，失败时返回错误
363    ///
364    /// # 线程安全
365    ///
366    /// 此方法是线程安全的，多个线程可以同时执行转换。
367    ///
368    /// # 示例
369    ///
370    /// ```rust
371    /// use syn::parse_quote;
372    /// use crate::converter::converter_registry::GlobalConverterRegistry;
373    ///
374    /// let field: syn::Field = parse_quote! {
375    ///     name: String
376    /// };
377    ///
378    /// let result = GlobalConverterRegistry::convert_field(&field);
379    /// assert!(result.is_ok());
380    /// ```
381    ///
382    /// # 设计原则体现
383    ///
384    /// - **门面模式**: 提供统一的转换接口
385    /// - **策略模式**: 内部选择合适的转换器
386    pub fn convert_field(field: &Field) -> MacroResult<TokenStream2> {
387        match GLOBAL_REGISTRY.read() {
388            Ok(registry) => registry.convert_field(field),
389            Err(e) => Err(MacroError::GenerationError {
390                message: format!("无法获取注册表读锁: {}", e),
391                span: None,
392            }),
393        }
394    }
395
396    /// 获取转换器数量
397    ///
398    /// 返回全局注册表中转换器的总数。
399    /// 主要用于调试和监控。
400    ///
401    /// # 返回值
402    ///
403    /// 成功时返回转换器数量，失败时返回锁定错误
404    ///
405    /// # 设计原则体现
406    ///
407    /// - **门面模式**: 提供统一的状态查询接口
408    pub fn converter_count() -> Result<usize, String> {
409        match GLOBAL_REGISTRY.read() {

413    }

415    /// 清空所有转换器
416    ///
417    /// 移除全局注册表中的所有转换器。
418    /// 主要用于测试环境的清理。
419    ///
420    /// # 返回值
421    ///
422    /// 成功时返回 Ok(())，失败时返回锁定错误
423    ///
424    /// # 警告
425    ///
426    /// 此操作会移除所有转换器，包括内置转换器。
427    /// 调用后需要重新注册转换器或调用 `reload_defaults`。
428    ///

430    ///

433        match GLOBAL_REGISTRY.write() {
434            Ok(mut registry) => {
435                registry.clear_converters();
436                Ok(())
437            }
438            Err(e) => Err(format!("无法获取注册表写锁: {}", e)),
439        }
440    }

442    /// 重新加载默认转换器
443    ///
444    /// 清空注册表并重新加载所有内置转换器。
445    /// 用于重置注册表到初始状态。
446    ///
447    /// # 返回值
448    ///
449    /// 成功时返回 Ok(())，失败时返回锁定错误
450    ///
451    /// # 使用场景
452    ///
453    /// - 测试后的清理
454    /// - 重置到初始状态
455    /// - 错误恢复
456    ///
457    /// # 设计原则体现
458    ///
459    /// - **工厂模式**: 重新创建默认转换器集合
460    pub fn reload_defaults() -> Result<(), String> {
461        match GLOBAL_REGISTRY.write() {
462            Ok(mut registry) => {
463                registry.reload_default_converters();
464                Ok(())
465            }
466            Err(e) => Err(format!("无法获取注册表写锁: {}", e)),
467        }
468    }

470    /// 检查是否支持指定类型
471    ///
472    /// 检查全局注册表中是否有支持指定类型的转换器。

476    ///

482    ///

484    ///
485    /// - **门面模式**: 提供统一的支持性检查接口
486    pub fn supports_type(field_type: &Type) -> Result<bool, String> {
487        match GLOBAL_REGISTRY.read() {
488            Ok(registry) => {
489                Ok(registry.find_converter_for_type(field_type).is_some())
490            }
491            Err(e) => Err(format!("无法获取注册表读锁: {}", e)),
492        }
493    }
494}
