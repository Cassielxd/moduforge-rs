1//! ModuForge-RS 宏扩展模块
2//!
3//! 提供 #[derive(Node)] 和 #[derive(Mark)] 派生宏，
4//! 支持声明式的节点和标记定义，简化 ModuForge-RS 框架的使用。
5//!
6//! # 功能特性
7//! 
8//! - **Node 派生宏**: 自动生成 `to_node()` 方法，将结构体转换为 mf_core::node::Node
9//! - **Mark 派生宏**: 自动生成 `to_mark()` 方法，将结构体转换为 mf_core::mark::Mark
10//! - **编译时验证**: 在编译期检查属性配置和类型兼容性
11//! - **友好错误消息**: 提供详细的编译错误信息和修复建议
12//!
13//! # 设计原则
14//!
15//! 此模块严格遵循 SOLID 设计原则：
16//! - **单一职责原则 (SRP)**: 每个模块只负责一个明确的功能
17//! - **接口隔离原则 (ISP)**: 提供精简、专用的接口
18//! - **开闭原则 (OCP)**: 通过插件系统支持扩展
19//! - **里氏替换原则 (LSP)**: 确保实现类型的可替换性
20

33/// 插件状态派生宏
34/// 
35/// 为结构体实现 Resource trait，用于依赖注入系统。
36/// 这是现有功能，保持不变以确保向后兼容性。
37#[proc_macro_derive(PState)]
38pub fn derive_plugin_state(input: TokenStream) -> TokenStream {
39    let input = parse_macro_input!(input as DeriveInput);
40    let name = input.ident;
41
42    let expanded = quote! {
43        impl Resource for #name {}
44    };
45
46    TokenStream::from(expanded)
47}

49/// Node 派生宏
50///
51/// 为结构体生成 `to_node()` 方法，将结构体实例转换为 `mf_core::node::Node`。
52/// 
53/// # 支持的属性
54/// 
55/// - `#[node_type = "类型名"]` - 必需，指定节点类型标识符
56/// - `#[marks = "mark1 mark2"]` - 可选，指定支持的标记类型列表
57/// - `#[content = "内容表达式"]` - 可选，指定内容约束表达式
58/// - `#[attr]` - 字段级属性，标记字段作为节点属性
59/// 
60/// # 示例
61/// 
62/// ```rust
63/// use mf_derive::Node;
64/// 
65/// #[derive(Node)]
66/// #[node_type = "project"]
67/// #[marks = "color bold"]
68/// #[content = "text*"]
69/// pub struct ProjectNode {
70///     #[attr]
71///     name: String,
72///     
73///     #[attr] 
74///     description: Option<String>,
75/// }
76/// 
77/// // 使用生成的方法
78/// let project = ProjectNode {
79///     name: "示例项目".to_string(),
80///     description: Some("这是一个示例项目".to_string()),
81/// };
82/// let node = project.to_node();
83/// ```
84/// 
85/// # 设计原则体现
86/// 
87/// - **单一职责**: 只负责 Node 相关的派生宏功能
88/// - **开闭原则**: 通过属性配置支持扩展，无需修改代码
89/// - **接口隔离**: 提供专门的 Node 转换接口
90#[proc_macro_derive(Node, attributes(node_type, marks, content, attr))]
91pub fn derive_node(input: TokenStream) -> TokenStream {
92    let input = parse_macro_input!(input as DeriveInput);
93    
94    // 使用带错误恢复的处理函数，确保友好的编译错误消息
95    let result = node::derive_impl::process_derive_node_with_recovery(input);
96    TokenStream::from(result)
97}

99/// Mark 派生宏
100/// 
101/// 为结构体生成 `to_mark()` 方法，将结构体实例转换为 `mf_core::mark::Mark`。
102/// 
103/// # 支持的属性
104/// 
105/// - `#[mark_type = "类型名"]` - 必需，指定标记类型标识符
106/// - `#[attr]` - 字段级属性，标记字段作为标记属性
107/// 
108/// # 示例
109/// 
110/// ```rust
111/// use mf_derive::Mark;
112/// 
113/// #[derive(Mark)]
114/// #[mark_type = "emphasis"]
115/// pub struct EmphasisMark {
116///     #[attr]
117///     level: i32,
118///     
119///     #[attr]
120///     color: Option<String>,
121/// }
122/// 
123/// // 使用生成的方法
124/// let emphasis = EmphasisMark {
125///     level: 2,
126///     color: Some("red".to_string()),
127/// };
128/// let mark = emphasis.to_mark();
129/// ```
130/// 
131/// # 设计原则体现
132/// 
133/// - **单一职责**: 只负责 Mark 相关的派生宏功能
134/// - **里氏替换原则**: 生成的 Mark 实例可完全替换手动创建的实例
135/// - **接口隔离**: 提供专门的 Mark 转换接口
136#[proc_macro_derive(Mark, attributes(mark_type, attr))]
137pub fn derive_mark(input: TokenStream) -> TokenStream {
138    let input = parse_macro_input!(input as DeriveInput);
139    
140    // 使用带错误恢复的处理函数，确保友好的编译错误消息
141    let result = mark::derive_impl::process_derive_mark_with_recovery(input);
142    TokenStream::from(result)
143}
