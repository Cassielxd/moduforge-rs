1//! Node 代码生成器实现
2//!
3//! 专门负责为 #[derive(Node)] 宏生成 node_definition()、from() 等方法的代码。
4//! 严格遵循单一职责原则，只负责 Node 相关的代码生成逻辑。
5//! 支持自定义类型表达式和泛型类型处理，要求自定义类型实现 Default + Serialize traits。

28/// Node 代码生成器
29///
30/// 负责为结构体生成 to_node() 方法，将结构体实例转换为 mf_core::node::Node。
31/// 遵循单一职责原则，专门处理 Node 相关的代码生成。
32///
33/// # 设计原则体现
34///
35/// - **单一职责原则**: 只负责 Node 代码生成，不处理其他类型
36/// - **开闭原则**: 通过配置和转换器扩展功能而不修改核心逻辑
37/// - **里氏替换原则**: 实现了 CodeGenerator trait，可以替换其他生成器使用
38#[derive(Debug)]
39pub struct NodeGenerator<'a> {
40    /// 派生宏的输入，包含结构体定义
41    input: &'a DeriveInput,
42    
43    /// Node 配置信息，包含所有解析后的属性
44    config: &'a NodeConfig,
45    
46}
47

50    ///
51    /// # 参数
52    ///
53    /// * `input` - 派生宏的输入，包含结构体定义
54    /// * `config` - Node 配置信息，包含解析后的所有属性
55    ///
56    /// # 返回值
57    ///
58    /// 返回配置好的 Node 代码生成器实例
59    ///
60    /// # 设计原则体现
61    ///
62    /// - **单一职责**: 只负责实例初始化
63    /// - **依赖注入**: 通过参数接收依赖的配置信息
64    ///
65    /// # 示例
66    ///
67    /// ```rust
68    /// use crate::generator::node_generator::NodeGenerator;
69    /// use crate::parser::{AttributeParser, NodeConfig};
70    /// use syn::parse_quote;
71    ///
72    /// let input = parse_quote! {
73    ///     #[derive(Node)]

75    ///     #[marks = "bold italic"]

77    ///     struct MyNode {
78    ///         #[attr]
79    ///         title: String,
80    ///         
81    ///         #[attr(default="Hello World")]
82    ///         content: String,
83    ///         
84    ///         #[attr(default="CustomStruct::new()")]
85    ///         custom_field: CustomStruct,  // 需要实现 Default + Serialize
86    ///         
87    ///         // 非属性字段，在 from() 中使用默认值，不出现在 node_definition() 中
88    ///         non_attr_field: String,
89    ///     }
90    /// };
91    ///
92    /// let config = AttributeParser::parse_node_attributes(&input).unwrap();
93    /// let generator = NodeGenerator::new(&input, &config);
94    /// 
95    /// // 生成的代码包含：
96    /// // - node_definition() 静态方法（只包含 #[attr] 字段的 AttributeSpec）
97    /// // - from() 方法（处理所有字段，支持类型验证和错误处理）
98    /// // - default_instance() 方法（失败时的降级方法）
99    /// // - From trait 实现（双向转换）
100    /// ```
101    pub fn new(input: &'a DeriveInput, config: &'a NodeConfig) -> Self {
102        Self {
103            input,
104            config,
105        }

108    /// 生成 node_definition() 方法的实现代码
109    ///
110    /// 根据配置信息生成完整的 node_definition() 方法实现。
111    /// 此方法是代码生成器的核心功能，返回节点定义而不是具体实例。
112    /// **重要**: 只为标记了 #[attr] 的字段生成属性定义。
113    ///
114    /// # 返回值
115    ///
116    /// 成功时返回生成的代码 TokenStream，失败时返回生成错误
117    ///
118    /// # 生成的方法结构
119    ///
120    /// ```rust
121    /// impl StructName {
122    ///     pub fn node_definition() -> mf_core::node::Node {
123    ///         // 导入必要的类型
124    ///         use mf_model::node_type::NodeSpec;
125    ///         use std::collections::HashMap;
126    ///         use serde_json::Value as JsonValue;
127    ///         
128    ///         // 只为 #[attr] 字段构建属性映射
129    ///         let mut attrs_map = std::collections::HashMap::new();
130    ///         // 支持自定义类型表达式 (需要实现 Default + Serialize)
131    ///         attrs_map.insert("field_name".to_string(), AttributeSpec {
132    ///             default: Some(serde_json::to_value(CustomType::new()).unwrap_or(null))
133    ///         });
134    ///         
135    ///         // 构建 NodeSpec
136    ///         let spec = NodeSpec { attrs: Some(attrs_map), ... };
137    ///         
138    ///         // 创建并返回 Node 定义
139    ///         mf_core::node::Node::create("node_type", spec)
140    ///     }
141    /// }
142    /// ```
143    ///
144    /// # 设计原则体现
145    ///
146    /// - **单一职责**: 只负责生成 node_definition() 方法代码
147    /// - **开闭原则**: 支持自定义类型表达式和转换器扩展
148    /// - **语义清晰**: 方法名明确表示返回的是定义而非实例
149    /// - **属性精确性**: 只包含 #[attr] 标记的字段，符合节点定义语义
150    pub fn generate_node_definition_method(&self) -> MacroResult<TokenStream2> {
151        let struct_name = &self.input.ident;
152        let node_type = self.config.node_type.as_ref()
153            .ok_or_else(|| MacroError::validation_error(
154                "Node 配置缺少必需的 node_type 属性",
155                self.input,
156            ))?;
157
158        // 生成必要的导入语句
159        let imports = self.generate_imports();
160
161        // 生成 NodeSpec 构建代码
162        let spec_code = self.generate_node_spec_code()?;
163
164        // 返回 node_definition 方法的实现，不包含 impl 块
165        let method_impl = quote! {
166            /// 获取节点定义
167            ///
168            /// 此方法由 #[derive(Node)] 宏自动生成，根据结构体的字段
169            /// 和宏属性配置创建节点定义（而非具体实例）。
170            ///
171            /// # 返回值
172            /// 
173            /// 返回配置好的 `mf_core::node::Node` 定义
174            ///
175            /// # 生成说明
176            ///
177            /// 这个方法是由 ModuForge-RS 宏系统自动生成的，
178            /// 它遵循以下设计原则：
179            /// - **单一职责**: 只负责 Node 定义的创建
180            /// - **语义清晰**: 方法名明确表示返回的是定义而非实例
181            /// - **里氏替换**: 生成的 Node 定义可以替换手动创建的定义
182            pub fn node_definition() -> mf_core::node::Node {
183                #imports
184                
185                #spec_code
186                
187                // 创建并返回 Node 定义
188                mf_core::node::Node::create(#node_type, spec)
189            }
190        };
191
192        Ok(method_impl)
193    }

213    /// 生成 NodeSpec 构建代码
214    ///
215    /// 根据配置信息生成 NodeSpec 的构建代码。
216    /// 遵循单一职责原则，专门负责 NodeSpec 的代码生成。
217    ///
218    /// # 返回值
219    ///
220    /// 成功时返回 NodeSpec 构建代码，失败时返回生成错误
221    ///
222    /// # 生成的代码结构
223    ///
224    /// ```rust
225    /// let spec = NodeSpec {
226    ///     content: Some("content_expression".to_string()),
227    ///     marks: Some("mark1,mark2".to_string()),
228    ///     attrs: attrs,
229    ///     group: None,
230    ///     desc: None,
231    /// };
232    /// ```
233    ///
234    /// # 设计原则体现
235    ///
236    /// - **单一职责**: 只负责 NodeSpec 构建代码生成
237    /// - **开闭原则**: 通过配置支持扩展而不修改代码
238    fn generate_node_spec_code(&self) -> MacroResult<TokenStream2> {
239        let content = self.config.content.as_ref().map(|c| quote! { Some(#c.to_string()) })
240            .unwrap_or_else(|| quote! { None });
241            
242        let marks = self.config.marks_string().map(|m| quote! { Some(#m.to_string()) })
243            .unwrap_or_else(|| quote! { None });
244
245        // 生成属性映射构建代码
246        let attrs_code = self.generate_attrs_spec_code()?;
247
248        let spec_code = quote! {
249            #attrs_code
250            
251            let spec = mf_model::node_type::NodeSpec {
252                content: #content,
253                marks: #marks,
254                attrs,
255                group: None,
256                desc: None,
257            };
258        };
259
260        Ok(spec_code)
261    }

263    /// 生成属性映射构建代码 (for NodeSpec)
264    ///
265    /// 只为添加了 #[attr] 标记的字段生成 NodeSpec 的属性映射构建代码。
266    /// 这是节点定义的核心逻辑，只有标记为属性的字段才会成为节点的属性。
267    /// 支持自定义类型表达式，要求实现 Default + Serialize traits。
268    ///
269    /// # 返回值
270    ///
271    /// 成功时返回属性映射构建代码，失败时返回生成错误
272    ///
273    /// # 生成的代码结构
274    ///
275    /// ```rust
276    /// let mut attrs_map = std::collections::HashMap::new();
277    /// 
278    /// // 基本类型默认值
279    /// attrs_map.insert("title".to_string(), AttributeSpec { 
280    ///     default: Some(serde_json::json!(String::default())) 
281    /// });
282    /// 
283    /// // 自定义类型表达式 (from #[attr(default="CustomType::new()")])
284    /// attrs_map.insert("custom_field".to_string(), AttributeSpec { 
285    ///     default: Some(serde_json::to_value(CustomType::new()).unwrap_or(serde_json::json!(null)))
286    /// });
287    /// 
288    /// // Option 类型
289    /// attrs_map.insert("optional_field".to_string(), AttributeSpec { 
290    ///     default: Some(serde_json::json!(null))
291    /// });
292    /// 
293    /// let attrs = Some(attrs_map);
294    /// ```
295    ///
296    /// # 设计原则体现
297    ///
298    /// - **单一职责**: 只负责属性映射代码生成
299    /// - **开闭原则**: 支持自定义类型表达式和新的字段类型扩展
300    /// - **精确性**: 只处理有 #[attr] 标记的字段，符合节点定义语义
301    /// - **类型安全**: 自定义类型必须实现 Default + Serialize
302    fn generate_attrs_spec_code(&self) -> MacroResult<TokenStream2> {
303        // 只获取有 #[attr] 标记的字段
304        let attr_fields = &self.config.attr_fields;
305        
306        if attr_fields.is_empty() {
307            // 没有属性字段时，创建空的 attrs
308            return Ok(quote! {
309                let attrs = None;
310            });
311        }
312
313        let mut field_setters = Vec::new();
314
315        // 为每个属性字段生成设置代码
316        for field_config in attr_fields {
317            let field_setter = self.generate_field_spec_code(field_config)?;
318            field_setters.push(field_setter);
319        }
320
321        // 生成完整的属性映射构建代码
322        let attrs_code = quote! {
323            let mut attrs_map = std::collections::HashMap::new();
324            #(#field_setters)*
325            let attrs = Some(attrs_map);
326        };
327
328        Ok(attrs_code)
329    }

331    /// 提取所有字段信息
332    ///
333    /// 从 DeriveInput 中提取所有字段，包括有和没有 #[attr] 标记的字段。
334    ///
335    /// # 返回值
336    ///
337    /// 返回包含字段信息的向量
338    fn extract_all_fields(&self) -> MacroResult<Vec<FieldInfo>> {
339        use syn::{Data, Fields};
340
341        let mut all_fields = Vec::new();
342
343        match &self.input.data {
344            Data::Struct(data_struct) => {
345                match &data_struct.fields {
346                    Fields::Named(fields_named) => {
347                        for field in &fields_named.named {
348                            if let Some(field_name) = &field.ident {
349                                // 检查是否是有 #[attr] 标记的字段
350                                let field_config = self.config.attr_fields.iter()
351                                    .find(|config| &config.name == &field_name.to_string());
352
353                                let field_info = FieldInfo {
354                                    name: field_name.to_string(),
355                                    type_name: self.extract_type_name(&field.ty),
356                                    config: field_config.cloned(),
357                                };
358
359                                all_fields.push(field_info);
360                            }
361                        }
362                    }
363                    Fields::Unnamed(_) => {
364                        return Err(MacroError::validation_error(
365                            "不支持元组结构体",
366                            self.input,
367                        ));
368                    }
369                    Fields::Unit => {
370                        // 单元结构体，没有字段
371                    }
372                }
373            }
374            _ => {
375                return Err(MacroError::validation_error(
376                    "只支持结构体类型",
377                    self.input,
378                ));
379            }
380        }
381
382        Ok(all_fields)
383    }

385    /// 从类型中提取完整类型名称（包含泛型参数）
386    ///
387    /// 递归解析类型结构，构建包含完整泛型信息的类型名称字符串。
388    /// 这对于正确处理 Option<String>、Vec<u8> 等泛型类型至关重要。
389    ///
390    /// # 参数
391    ///
392    /// * `ty` - syn::Type 类型引用
393    ///
394    /// # 返回值
395    ///
396    /// 包含完整泛型信息的类型名称字符串
397    ///
398    /// # 示例
399    ///
400    /// ```rust
401    /// // 基本类型
402    /// String -> "String"
403    /// i32 -> "i32"
404    /// 
405    /// // 泛型类型
406    /// Option<String> -> "Option<String>"
407    /// Vec<u8> -> "Vec<u8>"
408    /// HashMap<String, i32> -> "HashMap<String, i32>"
409    /// 
410    /// // 嵌套泛型
411    /// Option<Vec<String>> -> "Option<Vec<String>>"
412    /// 
413    /// // 模块路径
414    /// std::collections::HashMap<String, i32> -> "std::collections::HashMap<String, i32>"
415    /// ```
416    fn extract_type_name(&self, ty: &syn::Type) -> String {
417        use syn::{Type, TypePath, PathArguments, GenericArgument, AngleBracketedGenericArguments};
418
419        match ty {
420            Type::Path(TypePath { path, .. }) => {
421                // 构建完整的类型名称，包括泛型参数
422                let segments: Vec<String> = path.segments.iter().map(|seg| {
423                    let ident = seg.ident.to_string();
424                    match &seg.arguments {
425                        PathArguments::AngleBracketed(AngleBracketedGenericArguments { args, .. }) => {
426                            let type_args: Vec<String> = args.iter().map(|arg| {
427                                match arg {
428                                    GenericArgument::Type(inner_ty) => self.extract_type_name(inner_ty),
429                                    _ => "Unknown".to_string(),
430                                }
431                            }).collect();
432                            if type_args.is_empty() {
433                                ident
434                            } else {
435                                format!("{}<{}>", ident, type_args.join(", "))
436                            }
437                        }
438                        _ => ident,
439                    }
440                }).collect();
441                segments.join("::")
442            }
443            _ => "Unknown".to_string(),
444        }
445    }

447    /// 根据字段信息生成属性设置代码
448    ///
449    /// 根据字段信息生成相应的属性设置代码，所有字段都使用默认值：
450    /// 1. 如果有 #[attr] 标记且有 default 值，使用 default 值
451    /// 2. 如果有 #[attr] 标记但没有 default 值，使用字段类型的默认值
452    /// 3. 如果没有 #[attr] 标记，使用字段类型的默认值
453    ///
454    /// # 参数
455    ///
456    /// * `field_info` - 字段信息
457    ///
458    /// # 返回值
459    ///
460    /// 成功时返回字段属性设置代码，失败时返回生成错误
461    fn generate_field_spec_from_info(&self, field_info: &FieldInfo) -> MacroResult<TokenStream2> {
462        let field_name = &field_info.name;
463
464        // 生成默认值表达式
465        let default_value_expr = if let Some(config) = &field_info.config {
466            // 如果有 #[attr] 配置
467            if config.default_value.is_some() {
468                // 有 default 值，使用 default 值
469                self.generate_default_value_expression(config)?
470            } else {
471                // 没有 default 值，使用字段类型的默认值
472                self.generate_type_default_value(&field_info.type_name)?
473            }
474        } else {
475            // 没有 #[attr] 标记，使用字段类型的默认值
476            self.generate_type_default_value(&field_info.type_name)?
477        };
478
479        // 生成属性设置代码，创建 AttributeSpec
480        let attr_code = quote! {
481            attrs_map.insert(#field_name.to_string(), mf_model::schema::AttributeSpec {
482                default: Some(#default_value_expr)
483            });
484        };
485
486        Ok(attr_code)
487    }

513    /// });

519    /// - **里氏替换**: 对任何字段配置都能正确处理
520    /// - **开闭原则**: 支持 default 属性扩展而不修改核心逻辑
521    fn generate_field_spec_code(&self, field_config: &FieldConfig) -> MacroResult<TokenStream2> {
522        let field_name = &field_config.name;
523
524        // 生成默认值表达式
525        let default_value_expr = self.generate_default_value_expression(field_config)?;
526
527        // 生成属性设置代码，创建 AttributeSpec
528        let attr_code = quote! {
529            attrs_map.insert(#field_name.to_string(), mf_model::schema::AttributeSpec {
530                default: Some(#default_value_expr)
531            });
532        };
533
534        Ok(attr_code)

537    /// 生成字段的默认值表达式
538    ///
539    /// 根据字段配置生成相应的默认值表达式：
540    /// 1. 如果字段有 default 属性，使用该默认值
541    /// 2. 如果没有 default 属性，使用类型的默认值
542    ///
543    /// # 参数
544    ///
545    /// * `field_config` - 字段配置信息
546    ///
547    /// # 返回值
548    ///
549    /// 成功时返回默认值表达式代码，失败时返回生成错误
550    ///
551    /// # 设计原则体现
552    ///
553    /// - **单一职责**: 专门负责默认值表达式生成
554    /// - **开闭原则**: 支持新的默认值类型扩展
555    fn generate_default_value_expression(&self, field_config: &FieldConfig) -> MacroResult<TokenStream2> {
556        // 检查是否有 default 属性
557        if let Some(default_value) = &field_config.default_value {
558            // 使用 attr 中的 default 值
559            return self.generate_default_value_from_attr(default_value);
560        }
561
562        // 没有 default 属性时，使用类型的默认值
563        self.generate_type_default_value(&field_config.type_name)

572    /// * `default_value` - 默认值配置
573    ///
574    /// # 返回值

579
580        match &default_value.value_type {

583            }

589            }
590            DefaultValueType::Boolean(b) => {
591                Ok(quote! { serde_json::json!(#b) })
592            }
593            DefaultValueType::Json(json_value) => {

617    /// 生成类型的默认值表达式
618    ///
619    /// 为没有 default 属性的字段生成类型默认值的 JSON 表达式。
620    /// 支持泛型类型（如 Option<String>、Vec<u8>）和自定义类型。
621    /// 自定义类型必须实现 Default + Serialize traits。
622    ///
623    /// # 参数
624    ///
625    /// * `type_name` - 字段类型名称（支持泛型，如 "Option<String>"）
626    ///
627    /// # 返回值
628    ///
629    /// 返回类型默认值的 JSON 表达式代码
630    ///
631    /// # 支持的类型示例
632    ///
633    /// ```rust
634    /// // 基本类型
635    /// "String" => serde_json::json!(String::default())
636    /// "i32" => serde_json::json!(0)
637    /// "bool" => serde_json::json!(false)
638    /// 
639    /// // 泛型类型
640    /// "Option<String>" => serde_json::json!(null)
641    /// "Vec<u8>" => serde_json::json!(Vec::<u8>::new())
642    /// 
643    /// // 自定义类型 (需要 Default + Serialize)
644    /// "CustomStruct" => serde_json::to_value(<CustomStruct as Default>::default())
645    /// ```
646    fn generate_type_default_value(&self, type_name: &str) -> MacroResult<TokenStream2> {
647        let default_expr = match type_name {
648            "String" => quote! { serde_json::json!(String::default()) },
649            "i8" | "i16" | "i32" | "i64" | "i128" | "isize" => quote! { serde_json::json!(0) },
650            "u8" | "u16" | "u32" | "u64" | "u128" | "usize" => quote! { serde_json::json!(0) },
651            "f32" | "f64" => quote! { serde_json::json!(0.0) },
652            "bool" => quote! { serde_json::json!(false) },
653            "serde_json::Value" | "Value" => quote! { serde_json::json!(null) },
654            "uuid::Uuid" | "Uuid" => quote! { serde_json::json!(uuid::Uuid::new_v4().to_string()) },
655            "Vec<u8>" => quote! { serde_json::json!(Vec::<u8>::new()) },
656            "Vec<String>" => quote! { serde_json::json!(Vec::<String>::new()) },
657            _ if type_name.starts_with("Option<") => {
658                // Option 类型默认为 None，在 JSON 中表示为 null
659                quote! { serde_json::json!(null) }
660            },
661            _ => {
662                // 对于其他自定义类型，尝试使用 Default trait 并序列化
663                // 这要求类型实现 Default + Serialize traits
664                // 使用简单的字符串替换而不是解析复杂类型
665                if let Ok(type_ident) = syn::parse_str::<syn::Type>(type_name) {
666                    quote! { 
667                        serde_json::to_value(<#type_ident as Default>::default())
668                            .unwrap_or_else(|_| serde_json::json!(null))
669                    }
670                } else {
671                    // 如果类型解析失败，回退到 null
672                    quote! { serde_json::json!(null) }
673                }
674            }
675        };
676
677        Ok(default_expr)
678    }

680    /// 生成非 attr 字段的默认值
681    ///
682    /// 为没有 #[attr] 标记的字段生成默认值表达式。
683    /// 这些字段不会从 Node 的 attrs 中提取，而是使用类型默认值。
684    /// 支持泛型类型和自定义类型的类型安全处理。
685    ///
686    /// # 参数
687    ///
688    /// * `type_name` - 字段类型名称（支持泛型，如 "Option<String>"）
689    ///
690    /// # 返回值
691    ///
692    /// 返回默认值表达式代码
693    ///
694    /// # 设计原则
695    ///
696    /// 对于没有 #[attr] 标记的字段：
697    /// 1. 使用类型的默认值（如 String::default()）
698    /// 2. 对于自定义类型，使用类型安全的 <Type as Default>::default()
699    /// 3. 对于特殊类型（如 PhantomData），提供特定的处理
700    /// 4. 优雅降级：类型解析失败时使用通用 Default::default()
701    ///
702    /// # 支持的类型示例
703    ///
704    /// ```rust
705    /// // 基本类型
706    /// "String" => String::default()
707    /// "i32" => 0
708    /// "bool" => false
709    /// 
710    /// // 泛型类型
711    /// "Option<String>" => None
712    /// "Vec<u8>" => Vec::new()
713    /// 
714    /// // 自定义类型 (需要 Default trait)
715    /// "CustomStruct" => <CustomStruct as Default>::default()
716    /// 
717    /// // 特殊类型
718    /// "PhantomData<T>" => std::marker::PhantomData
719    /// "uuid::Uuid" => uuid::Uuid::new_v4()
720    /// ```
721    fn generate_non_attr_field_default(&self, type_name: &str) -> MacroResult<TokenStream2> {
722        let default_expr = match type_name {
723            "String" => quote! { String::default() },
724            "i8" | "i16" | "i32" | "i64" | "i128" | "isize" => quote! { 0 },
725            "u8" | "u16" | "u32" | "u64" | "u128" | "usize" => quote! { 0 },
726            "f32" | "f64" => quote! { 0.0 },
727            "bool" => quote! { false },
728            "uuid::Uuid" | "Uuid" => quote! { uuid::Uuid::new_v4() },
729            "Vec<u8>" => quote! { Vec::new() },
730            "Vec<String>" => quote! { Vec::new() },
731            _ if type_name.starts_with("Option<") => {
732                quote! { None } 
733            },
734            _ if type_name.contains("PhantomData") => {
735                quote! { std::marker::PhantomData }
736            },
737            _ => {
738                // 对于自定义类型，尝试使用 Default trait
739                // 如果该类型没有实现 Default，编译时会报错，这是预期的行为
740                if let Ok(type_ident) = syn::parse_str::<syn::Type>(type_name) {
741                    quote! { <#type_ident as Default>::default() }
742                } else {
743                    // 如果类型解析失败，使用通用 Default
744                    quote! { Default::default() }
745                }
746            }
747        };
748
749        Ok(default_expr)
750    }

752    /// 生成 default_instance 方法的实现代码
753    ///
754    /// 生成一个创建默认实例的方法，当 From 转换失败时使用。
755    ///
756    /// # 返回值
757    ///
758    /// 成功时返回生成的代码 TokenStream，失败时返回生成错误
759    pub fn generate_default_instance_method(&self) -> MacroResult<TokenStream2> {
760        // 获取所有字段信息
761        let all_fields = self.extract_all_fields()?;
762        
763        let mut field_inits = Vec::new();
764        
765        for field_info in all_fields {

767                .map_err(|_| MacroError::parse_error(

777                    self.generate_type_default_for_instance(&field_info.type_name)?
778                }
779            } else {
780                self.generate_type_default_for_instance(&field_info.type_name)?
781            };
782            
783            field_inits.push(quote! {
784                #field_name: #default_value
785            });
786        }
787

799                    #(#field_inits),*
800                }
801            }
802        };
803        
804        Ok(method_impl)
805    }

903    /// 生成 from 方法的实现代码
904    ///
905    /// 根据配置信息生成 from 方法，该方法接受 mf_model::node::Node 参数
906    /// 并返回当前结构体的实例。处理所有字段（包括有和没有 #[attr] 标记的）。
907    ///
908    /// # 返回值
909    ///
910    /// 成功时返回生成的代码 TokenStream，失败时返回生成错误
911    ///
912    /// # 生成的方法结构
913    ///
914    /// ```rust
915    /// impl StructName {
916    ///     pub fn from(node: &mf_model::node::Node) -> Result<Self, String> {
917    ///         use serde_json::Value as JsonValue;
918    ///         
919    ///         // 验证节点类型匹配
920    ///         if node.r#type != "expected_type" {
921    ///             return Err(format!("节点类型不匹配: 期望 '{}', 实际 '{}'", 
922    ///                 "expected_type", node.r#type));
923    ///         }
924    ///         
925    ///         Ok(Self {
926    ///             // #[attr] 字段：从 node.attrs 中提取
927    ///             attr_field: node.attrs.attrs.get("attr_field")
928    ///                 .and_then(|v| v.as_str())
929    ///                 .map(|s| s.to_string())
930    ///                 .unwrap_or_default(),
931    ///                 
932    ///             // 非 #[attr] 字段：使用默认值
933    ///             non_attr_field: <FieldType as Default>::default(),
934    ///             
935    ///             // Option 类型处理

937    ///                 .and_then(|v| v.as_str())
938    ///                 .map(|s| s.to_string()),
939    ///                 
940    ///             // 自定义类型处理
941    ///             custom_field: node.attrs.attrs.get("custom_field")

953    /// - **字段分离**: #[attr] 字段从 attrs 提取，非 #[attr] 字段使用默认值
954    /// - **类型安全**: 支持泛型类型和自定义类型的安全转换
955    pub fn generate_from_method(&self) -> MacroResult<TokenStream2> {

961            ))?;
962
963        // 生成字段初始化代码
964        let field_inits = self.generate_field_initializers()?;
965
966        // 只返回 from 方法的实现，不包含 impl 块
967        let method_impl = quote! {
968            /// 从 mf_model::node::Node 创建结构体实例
969            ///
970            /// 此方法由 #[derive(Node)] 宏自动生成，根据 Node 的属性
971            /// 创建相应的结构体实例。
972            ///
973            /// # 参数
974            ///
975            /// * `node` - 要转换的 Node 实例
976            ///
977            /// # 返回值
978            /// 
979            /// 成功时返回结构体实例，失败时返回错误信息
980            ///
981            /// # 错误
982            ///
983            /// 当节点类型不匹配时，返回包含错误信息的 Result
984            ///
985            /// # 生成说明
986            ///
987            /// 这个方法是由 ModuForge-RS 宏系统自动生成的，
988            /// 它遵循以下设计原则：
989            /// - **单一职责**: 只负责从 Node 创建结构体实例
990            /// - **错误安全**: 使用 Result 类型处理类型不匹配错误
991            pub fn from(node: &mf_model::node::Node) -> Result<Self, String> {
992                use serde_json::Value as JsonValue;
993                
994                // 验证节点类型匹配
995                if node.r#type != #node_type {
996                    return Err(format!("节点类型不匹配: 期望 '{}', 实际 '{}'", #node_type, node.r#type));
997                }
998                
999                Ok(Self {
1000                    #field_inits
1001                })
1002            }
1003        };
1004
1005        Ok(method_impl)
1006    }

1081            ))?;
1082
1083        // 根据字段类型生成不同的提取逻辑
1084        let extraction_code = self.generate_field_extraction_code(field_config)?;
1085
1086        Ok(quote! {
1087            #field_ident: #extraction_code
1088        })
1089    }
1090
1091    /// 生成字段值提取代码
1092    ///
1093    /// 根据字段类型生成相应的值提取和转换代码。
1094    ///
1095    /// # 参数
1096    ///
1097    /// * `field_config` - 字段配置信息
1098    ///
1099    /// # 返回值
1100    ///
1101    /// 成功时返回字段值提取代码，失败时返回转换错误
1102    fn generate_field_extraction_code(&self, field_config: &FieldConfig) -> MacroResult<TokenStream2> {
1103        let field_name = &field_config.name;
1104        let type_name = &field_config.type_name;
1105
1106        // 为不同类型生成不同的提取逻辑
1107        let extraction = match type_name.as_str() {
1108            "String" => quote! {
1109                node.attrs.attrs.get(#field_name)
1110                    .and_then(|v| v.as_str())
1111                    .map(|s| s.to_string())
1112                    .unwrap_or_default()
1113            },
1114            "i32" => quote! {
1115                node.attrs.attrs.get(#field_name)
1116                    .and_then(|v| v.as_i64())
1117                    .map(|i| i as i32)
1118                    .unwrap_or_default()
1119            },
1120            "f64" => quote! {
1121                node.attrs.attrs.get(#field_name)
1122                    .and_then(|v| v.as_f64())
1123                    .unwrap_or_default()
1124            },
1125            "bool" => quote! {
1126                node.attrs.attrs.get(#field_name)
1127                    .and_then(|v| v.as_bool())
1128                    .unwrap_or_default()
1129            },
1130            "serde_json::Value" | "Value" => quote! {
1131                node.attrs.attrs.get(#field_name)
1132                    .cloned()
1133                    .unwrap_or(serde_json::Value::Null)
1134            },
1135            "uuid::Uuid" | "Uuid" => quote! {
1136                node.attrs.attrs.get(#field_name)
1137                    .and_then(|v| v.as_str())
1138                    .and_then(|s| uuid::Uuid::parse_str(s).ok())
1139                    .unwrap_or_else(uuid::Uuid::new_v4)
1140            },
1141            "Vec<u8>" => quote! {
1142                node.attrs.attrs.get(#field_name)
1143                    .and_then(|v| v.as_array())
1144                    .map(|arr| arr.iter().filter_map(|v| v.as_u64().map(|u| u as u8)).collect())
1145                    .unwrap_or_default()
1146            },
1147            "Vec<String>" => quote! {
1148                node.attrs.attrs.get(#field_name)
1149                    .and_then(|v| v.as_array())
1150                    .map(|arr| arr.iter().filter_map(|v| v.as_str().map(|s| s.to_string())).collect())
1151                    .unwrap_or_default()
1152            },
1153            _ if type_name.starts_with("Option<") => {
1154                // 处理 Option 类型
1155                let inner_type = self.extract_option_inner_type(type_name);
1156                match inner_type.as_str() {
1157                    "String" => quote! {
1158                        node.attrs.attrs.get(#field_name)
1159                            .and_then(|v| v.as_str())
1160                            .map(|s| s.to_string())
1161                    },
1162                    "i32" => quote! {
1163                        node.attrs.attrs.get(#field_name)
1164                            .and_then(|v| v.as_i64())
1165                            .map(|i| i as i32)
1166                    },
1167                    "f64" => quote! {
1168                        node.attrs.attrs.get(#field_name)
1169                            .and_then(|v| v.as_f64())
1170                    },
1171                    "bool" => quote! {
1172                        node.attrs.attrs.get(#field_name)
1173                            .and_then(|v| v.as_bool())
1174                    },
1175                    "serde_json::Value" | "Value" => quote! {
1176                        node.attrs.attrs.get(#field_name).cloned()
1177                    },
1178                    "uuid::Uuid" | "Uuid" => quote! {
1179                        node.attrs.attrs.get(#field_name)
1180                            .and_then(|v| v.as_str())
1181                            .and_then(|s| uuid::Uuid::parse_str(s).ok())
1182                    },
1183                    _ => quote! {
1184                        None
1185                    }
1186                }
1187            },
1188            _ => {
1189                return Err(MacroError::validation_error(
1190                    &format!("不支持的字段类型: {}", type_name),
1191                    self.input,
1192                ));
1193            }
1194        };
1195
1196        Ok(extraction)
1197    }
1198
1199    /// 提取 Option<T> 类型的内部类型
1200    ///
1201    /// 从 "Option<T>" 字符串中提取 "T" 部分。
1202    ///
1203    /// # 参数
1204    ///
1205    /// * `type_name` - 类型名称字符串
1206    ///
1207    /// # 返回值
1208    ///
1209    /// 返回内部类型名称
1210    fn extract_option_inner_type(&self, type_name: &str) -> String {
1211        if let Some(start) = type_name.find('<') {
1212            if let Some(end) = type_name.rfind('>') {
1213                if start < end {
1214                    return type_name[start + 1..end].to_string();
1215                }
1216            }
1217        }
1218        "String".to_string() // 默认返回 String
1219    }
1220}

1222impl<'a> CodeGenerator for NodeGenerator<'a> {
1223    /// 生成完整的 Node 代码
1224    ///
1225    /// 实现 CodeGenerator trait 的核心方法，生成完整的 Node 转换代码。
1226    /// 
1227    /// # 返回值
1228    ///
1229    /// 成功时返回生成的代码 TokenStream，失败时返回生成错误
1230    ///
1231    /// # 设计原则体现
1232    ///
1233    /// - **里氏替换原则**: 完全符合 CodeGenerator 接口契约
1234    /// - **单一职责**: 委托给专门的方法处理具体生成逻辑
1235    fn generate(&self) -> MacroResult<TokenStream2> {
1236        let struct_name = &self.input.ident;
1237        let node_definition_method = self.generate_node_definition_method()?;
1238        let from_method = self.generate_from_method()?;
1239        let default_instance_method = self.generate_default_instance_method()?;
1240        
1241        Ok(quote! {
1242            impl #struct_name {
1243                #node_definition_method
1244                
1245                #from_method
1246                
1247                #default_instance_method
1248            }
1249            
1250            impl From<#struct_name> for mf_core::node::Node {
1251                /// 将结构体实例转换为 mf_core::node::Node
1252                ///
1253                /// 实现标准的 From trait，支持使用 `.into()` 方法进行转换。
1254                /// 此实现由 #[derive(Node)] 宏自动生成。
1255                ///
1256                /// # 参数
1257                ///
1258                /// * `_value` - 结构体实例（当前实现中使用定义而非实例值）
1259                ///
1260                /// # 返回值
1261                ///
1262                /// 返回配置好的 `mf_core::node::Node` 定义
1263                ///
1264                /// # 使用示例
1265                ///
1266                /// ```rust
1267                /// let my_struct = MyStruct { /* fields */ };
1268                /// let node: mf_core::node::Node = my_struct.into();
1269                /// // 或者
1270                /// let node = mf_core::node::Node::from(my_struct);
1271                /// ```
1272                fn from(_value: #struct_name) -> Self {
1273                    #struct_name::node_definition()
1274                }
1275            }
1276            
1277            impl From<mf_model::node::Node> for #struct_name {
1278                /// 从 mf_model::node::Node 转换为结构体实例
1279                ///
1280                /// 实现标准的 From trait，支持使用 `.into()` 方法进行反向转换。
1281                /// 此实现由 #[derive(Node)] 宏自动生成。
1282                ///
1283                /// # 参数
1284                ///
1285                /// * `node` - mf_model::node::Node 实例
1286                ///
1287                /// # 返回值
1288                ///
1289                /// 返回结构体实例，如果转换失败则使用默认值
1290                ///
1291                /// # 使用示例
1292                ///
1293                /// ```rust
1294                /// let node: mf_model::node::Node = /* ... */;
1295                /// let my_struct: MyStruct = node.into();
1296                /// // 或者
1297                /// let my_struct = MyStruct::from(node);
1298                /// ```
1299                fn from(node: mf_model::node::Node) -> Self {
1300                    #struct_name::from(&node).unwrap_or_else(|_| {
1301                        // 如果转换失败，使用默认值创建实例
1302                        Self::default_instance()
1303                    })
1304                }
1305            }
1306        })
1307    }
1308
1309    /// 获取生成器名称
1310    ///
1311    /// 返回 Node 代码生成器的名称，用于调试和错误消息。
1312    ///
1313    /// # 返回值
1314    ///
1315    /// 返回生成器名称 "NodeGenerator"
1316    ///
1317    /// # 设计原则体现
1318    ///
1319    /// - **接口隔离**: 提供清晰的生成器标识
1320    fn name(&self) -> &'static str {
1321        "NodeGenerator"
1322    }
1323}
