72/// 同步事件
73#[derive(Debug, Clone)]
74pub enum SyncEvent {
75    /// 协议同步状态变化
76    ProtocolStateChanged(ProtocolSyncState),
77    /// 首次同步完成（空房间也算）
78    InitialSyncCompleted { has_data: bool, elapsed_ms: u64 },
79    /// 收到数据更新
80    DataReceived,
81    /// 连接状态变化
82    ConnectionChanged(ConnectionStatus),
83    /// 连接失败
84    ConnectionFailed(ConnectionError),
85}
