34    /// 注册转换器（编译时调用）
35    pub fn register_converter<T, C>(&mut self) -> &mut Self
36    where
37        T: Step + 'static,
38        C: super::typed_converter::TypedStepConverter<T> + Default + 'static,
39    {
40        let type_id = TypeId::of::<T>();
41        let converter = Arc::new(ErasedConverter::new::<T, C>());
42
43        // 检查重复注册
44        if self.converters.contains_key(&type_id) {
45            tracing::warn!(
46                "转换器重复注册: {} for type {}",
47                converter.converter_name(),
48                converter.type_name()
49            );
50            return self;
51        }
52
53        // 存储转换器信息
54        let info = ConverterInfo {
55            type_name: converter.type_name(),
56            converter_name: converter.converter_name(),
57            priority: converter.priority(),
58            supports_concurrent: converter.supports_concurrent(),
59            step_type_id: type_id,
60        };
61
62        self.converter_info.insert(type_id, info);
63        self.converters.insert(type_id, converter.clone());
64
65        // 按优先级插入有序列表
66        let insert_pos = self
67            .ordered_converters
68            .iter()
69            .position(|c| c.priority() > converter.priority())
70            .unwrap_or(self.ordered_converters.len());
71
72        self.ordered_converters.insert(insert_pos, converter);
73
74        tracing::info!(
75            "✅ 转换器已注册: {} for {} (优先级: {})",
76            C::converter_name(),
77            std::any::type_name::<T>(),
78            C::priority()
79        );
80
81        self
82    }

84    /// 查找并应用转换器 - 主要性能路径
85    pub fn convert_step(
86        &self,
87        step: &dyn Step,
88        txn: &mut TransactionMut,
89        context: &ConversionContext,
90    ) -> ConversionResult<StepResult> {
91        let start_time = std::time::Instant::now();
92        let step_type_id = step.type_id();
93
94        // O(1) 精确匹配查找
95        if let Some(converter) = self.converters.get(&step_type_id) {
96            let result = converter.try_convert(step, txn, context);
97
98            // 更新性能统计
99            self.performance_stats.record_conversion(
100                step_type_id,
101                start_time.elapsed(),
102                result.is_ok(),
103            );
104
105            return result;
106        }
107
108        // 如果没有找到精确匹配，尝试有序列表（用于兼容性）
109        for converter in &self.ordered_converters {
110            if converter.type_id() == step_type_id {
111                let result = converter.try_convert(step, txn, context);
112                self.performance_stats.record_conversion(
113                    step_type_id,
114                    start_time.elapsed(),
115                    result.is_ok(),
116                );
117                return result;
118            }
119        }
120
121        // 没有找到任何匹配的转换器
122        self.performance_stats.record_conversion(
123            step_type_id,
124            start_time.elapsed(),
125            false,
126        );
127
128        Err(ConversionError::UnsupportedStepType {
129            step_type: step.name().to_string(),
130            type_id: step_type_id,
131        })
132    }

134    /// 批量转换步骤 - 优化的批处理路径
135    pub fn convert_steps_batch(
136        &self,
137        steps: &[&dyn Step],
138        txn: &mut TransactionMut,
139        context: &ConversionContext,
140    ) -> Vec<ConversionResult<StepResult>> {
141        let mut results = Vec::with_capacity(steps.len());
142
143        // 按类型对步骤进行分组以提高缓存效率
144        let mut grouped_steps: HashMap<TypeId, Vec<&dyn Step>> = HashMap::new();
145        for step in steps {
146            grouped_steps
147                .entry(step.type_id())
148                .or_insert_with(Vec::new)
149                .push(*step);
150        }
151
152        // 按组处理步骤
153        for (type_id, group_steps) in grouped_steps {
154            if let Some(converter) = self.converters.get(&type_id) {
155                for step in group_steps {
156                    let result = converter.try_convert(step, txn, context);
157                    results.push(result);
158                }
159            } else {
160                // 为未找到转换器的步骤添加错误
161                for step in group_steps {
162                    results.push(Err(ConversionError::UnsupportedStepType {
163                        step_type: step.name().to_string(),
164                        type_id,
165                    }));
166                }
167            }
168        }
169
170        results
171    }

238/// 性能统计信息
239#[derive(Debug)]
240pub struct PerformanceStats {
241    /// 总转换次数
242    total_conversions: std::sync::atomic::AtomicU64,
243    /// 成功转换次数
244    successful_conversions: std::sync::atomic::AtomicU64,
245    /// 按类型的转换统计
246    type_stats: RwLock<HashMap<TypeId, TypeConversionStats>>,
247    /// 创建时间
248    created_at: std::time::Instant,
249}
250
251#[derive(Debug, Clone)]
252pub struct TypeConversionStats {
253    pub total_count: u64,
254    pub success_count: u64,
255    pub total_duration: std::time::Duration,
256    pub avg_duration: std::time::Duration,
257    pub min_duration: std::time::Duration,
258    pub max_duration: std::time::Duration,
259}
260
261impl PerformanceStats {
262    pub fn new() -> Self {
263        Self {
264            total_conversions: std::sync::atomic::AtomicU64::new(0),
265            successful_conversions: std::sync::atomic::AtomicU64::new(0),
266            type_stats: RwLock::new(HashMap::new()),
267            created_at: std::time::Instant::now(),
268        }
269    }
270
271    pub fn record_conversion(
272        &self,
273        type_id: TypeId,
274        duration: std::time::Duration,
275        success: bool,
276    ) {
277        use std::sync::atomic::Ordering;
278
279        // 更新全局统计
280        self.total_conversions.fetch_add(1, Ordering::Relaxed);
281        if success {
282            self.successful_conversions.fetch_add(1, Ordering::Relaxed);
283        }
284
285        // 更新类型特定统计
286        let mut type_stats = self.type_stats.write().unwrap();
287        let stats =
288            type_stats.entry(type_id).or_insert_with(|| TypeConversionStats {
289                total_count: 0,
290                success_count: 0,
291                total_duration: std::time::Duration::ZERO,
292                avg_duration: std::time::Duration::ZERO,
293                min_duration: std::time::Duration::MAX,
294                max_duration: std::time::Duration::ZERO,
295            });
296
297        stats.total_count += 1;
298        if success {
299            stats.success_count += 1;
300        }
301
302        stats.total_duration += duration;
303        stats.avg_duration = stats.total_duration / stats.total_count as u32;
304        stats.min_duration = stats.min_duration.min(duration);
305        stats.max_duration = stats.max_duration.max(duration);
306    }
307
308    pub fn get_total_conversions(&self) -> u64 {
309        self.total_conversions.load(std::sync::atomic::Ordering::Relaxed)
310    }
311
312    pub fn get_success_rate(&self) -> f64 {
313        let total = self.get_total_conversions();
314        if total == 0 {
315            0.0
316        } else {
317            let successful = self
318                .successful_conversions
319                .load(std::sync::atomic::Ordering::Relaxed);
320            successful as f64 / total as f64
321        }
322    }
323
324    pub fn get_type_stats(
325        &self,
326        type_id: TypeId,
327    ) -> Option<TypeConversionStats> {
328        self.type_stats.read().unwrap().get(&type_id).cloned()
329    }
330
331    pub fn get_uptime(&self) -> std::time::Duration {
332        self.created_at.elapsed()
333    }
334}
