10/// 根据已解析的扩展列表生成完整的 Schema
11///
12/// # 参数
13/// * `extensions` - 扩展列表，包含节点(N)、标记(M)和扩展(E)三种类型
14///
15/// # 返回值
16/// * `EditorResult<Schema>` - 返回编译后的 Schema 或错误
17///
18/// # 功能说明
19/// 1. 收集所有扩展中定义的全局属性
20/// 2. 处理节点扩展，构建节点定义
21/// 3. 处理标记扩展，构建标记定义
22/// 4. 确定顶层节点名称
23/// 5. 编译生成最终的 Schema
24pub fn get_schema_by_resolved_extensions(
25    extensions: &Vec<Extensions>
26) -> ForgeResult<Schema> {
27    // 收集所有扩展中定义的全局属性
28    let mut extension_attributes = vec![];
29    for extension in extensions {
30        if let Extensions::E(extension) = extension {
31            for item in extension.get_global_attributes().iter() {
32                extension_attributes.push(item);
33            }
34        }
35    }
36
37    // 初始化节点和标记的存储
38    let mut nodes = HashMap::new();
39    let mut marks = HashMap::new();
40    let mut top_name = "doc".to_string();
41
42    // 处理每个扩展
43    for extension in extensions {
44        match extension {
45            // 处理节点扩展
46            Extensions::N(node) => {
47                let name = node.name.clone();
48                // 检查是否为顶层节点
49                if node.is_top_node() {
50                    top_name = node.name.clone();
51                }
52                // 获取节点的属性定义
53                let mut attrs = get_attr_dfn(name, &extension_attributes);
54
55                // 合并节点类型中定义的属性
56                let attrs_def = match &node.r#type.attrs {
57                    Some(m) => {
58                        m.iter().for_each(|e| {
59                            attrs.insert(e.0.clone(), e.1.clone());
60                        });
61                        attrs
62                    },
63                    None => attrs,
64                };
65                let mut t = node.r#type.clone();
66                t.attrs = Some(attrs_def);
67                nodes.insert(node.name.clone(), t);
68            },
69            // 处理标记扩展
70            Extensions::M(mark) => {
71                marks.insert(mark.name.clone(), mark.r#type.clone());
72            },
73            _ => {},
74        }
75    }
76
77    // 创建 Schema 规范并编译
78    let instance_spec = SchemaSpec { nodes, marks, top_node: Some(top_name) };
79    let schema = Schema::compile(instance_spec)?;
80    Ok(schema)
81}

83/// 获取指定节点名称的属性定义
84///
85/// # 参数
86/// * `name` - 节点名称
87/// * `extension_attributes` - 全局属性列表
88///
89/// # 返回值
90/// * `HashMap<String, AttributeSpec>` - 节点对应的属性定义映射
91fn get_attr_dfn(
92    name: String,
93    extension_attributes: &Vec<&GlobalAttributeItem>,
94) -> HashMap<String, AttributeSpec> {
95    let mut attributes: HashMap<String, AttributeSpec> = HashMap::new();
96    // 遍历全局属性，找出适用于当前节点的属性
97    for attr in extension_attributes.iter() {
98        if attr.types.contains(&name) {
99            attr.attributes.iter().for_each(|e| {
100                attributes.insert(e.0.clone(), e.1.clone());
101            });
102        }
103    }
104    attributes
105}
