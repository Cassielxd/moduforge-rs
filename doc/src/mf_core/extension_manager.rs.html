152    /// 构建ExtensionManager
153    ///
154    /// # 返回值
155    /// * `ForgeResult<ExtensionManager>` - 构建的ExtensionManager实例或错误
156    pub fn build(self) -> ForgeResult<ExtensionManager> {
157        let start_time = Instant::now();
158        let mut all_extensions = self.extensions;
159
160        // 解析XML文件
161        for xml_file in &self.xml_files {
162            let extensions =
163                XmlSchemaParser::parse_multi_file_to_extensions(xml_file)
164                    .map_err(|e| {
165                        crate::error::error_utils::config_error(format!(
166                            "解析XML文件 {} 失败: {}",
167                            xml_file, e
168                        ))
169                    })?;
170            all_extensions.extend(extensions);
171        }
172
173        // 解析XML内容
174        for xml_content in &self.xml_contents {
175            let extensions = XmlSchemaParser::parse_to_extensions(xml_content)
176                .map_err(|e| {
177                    crate::error::error_utils::config_error(format!(
178                        "解析XML内容失败: {}",
179                        e
180                    ))
181                })?;
182            all_extensions.extend(extensions);
183        }
184
185        metrics::xml_parsing_duration(start_time.elapsed());
186
187        // 创建ExtensionManager
188        ExtensionManager::new(&all_extensions)
189    }

207    pub fn new(extensions: &Vec<Extensions>) -> ForgeResult<Self> {
208        let start_time = Instant::now();
209        let schema = Arc::new(get_schema_by_resolved_extensions(extensions)?);
210        let mut plugins = vec![];
211        let mut op_fns = vec![];
212        let mut extension_count = 0;
213        let mut plugin_count = 0;
214        for extension in extensions {
215            if let Extensions::E(extension) = extension {
216                extension_count += 1;
217                for plugin in extension.get_plugins() {
218                    plugin_count += 1;
219                    plugins.push(plugin.clone());
220                }
221                for op_fn in extension.get_op_fns() {
222                    op_fns.push(op_fn.clone());
223                }
224            }
225        }
226
227        metrics::extensions_loaded(extension_count);
228        metrics::plugins_loaded(plugin_count);
229        metrics::extension_manager_creation_duration(start_time.elapsed());
230
231        Ok(ExtensionManager { schema, plugins, op_fns })
232    }

350    /// 动态添加扩展并重新构建schema
351    ///
352    /// # 参数
353    /// * `new_extensions` - 要添加的新扩展列表
354    ///
355    /// # 返回值
356    /// * `ForgeResult<()>` - 成功或错误
357    ///
358    /// # 示例
359    /// ```rust
360    /// use mf_core::{ExtensionManager, types::Extensions, node::Node};
361    /// use mf_model::node_type::NodeSpec;
362    ///
363    /// let mut manager = ExtensionManager::new(&vec![])?;
364    ///
365    /// let new_node = Node::create("dynamic_node", NodeSpec::default());
366    /// manager.add_extensions(vec![Extensions::N(new_node)])?;
367    /// ```
368    pub fn add_extensions(
369        &mut self,
370        new_extensions: Vec<Extensions>,
371    ) -> ForgeResult<()> {
372        // 获取当前所有扩展
373        let mut all_extensions = Vec::new();
374
375        // 从当前schema重建扩展列表（这是一个简化的实现）
376        // 在实际应用中，你可能需要保存原始的扩展列表
377        for (name, node_type) in &self.schema.nodes {
378            let node = crate::node::Node::create(name, node_type.spec.clone());
379            all_extensions.push(Extensions::N(node));
380        }
381
382        for (name, mark_type) in &self.schema.marks {
383            let mark = crate::mark::Mark::new(name, mark_type.spec.clone());
384            all_extensions.push(Extensions::M(mark));
385        }
386
387        // 添加新扩展
388        all_extensions.extend(new_extensions);
389
390        // 重新构建ExtensionManager
391        let new_manager = Self::new(&all_extensions)?;
392
393        // 更新当前实例
394        self.schema = new_manager.schema;
395        self.plugins = new_manager.plugins;
396        self.op_fns = new_manager.op_fns;
397
398        Ok(())
399    }
