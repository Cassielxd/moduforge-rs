45#[derive(Clone, Default)]
46pub struct RuntimeOptions {
47    content: Content,
48    extensions: Vec<Extensions>,
49    history_limit: Option<usize>,
50    event_handlers: Vec<Arc<dyn EventHandler<Event> + Send + Sync>>,
51    middleware_stack: MiddlewareStack,
52}
53impl RuntimeOptions {
54    /// 从ExtensionManager创建RuntimeOptions
55    ///
56    /// # 参数
57    /// * `extension_manager` - ExtensionManager实例
58    ///
59    /// # 返回值
60    /// * `Self` - 新的RuntimeOptions实例
61    pub fn from_extension_manager(
62        extension_manager: crate::extension_manager::ExtensionManager
63    ) -> Self {
64        // 从ExtensionManager获取schema并重建extensions
65        let schema = extension_manager.get_schema();
66        let mut extensions = Vec::new();
67
68        // 重建节点扩展
69        for (name, node_type) in &schema.nodes {
70            let node = crate::node::Node::create(name, node_type.spec.clone());
71            extensions.push(Extensions::N(node));
72        }
73
74        // 重建标记扩展
75        for (name, mark_type) in &schema.marks {
76            let mark = crate::mark::Mark::new(name, mark_type.spec.clone());
77            extensions.push(Extensions::M(mark));
78        }
79
80        Self {
81            content: Content::None,
82            extensions,
83            history_limit: None,
84            event_handlers: Vec::new(),
85            middleware_stack: MiddlewareStack::default(),
86        }
87    }
