8/// 核心快照数据结构
9///
10/// 包含运行时需要的所有预编译数据，避免重复初始化
11#[derive(Debug, Clone, Serialize, Deserialize)]
12pub struct CoreSnapshot {
13    /// 快照版本，用于兼容性检查
14    pub version: String,
15    /// 快照创建时间戳
16    pub created_at: u64,
17    /// 预编译的配置
18    pub config: ForgeConfig,
19    /// 序列化的 Schema (使用 JSON 格式)
20    pub schema_json: String,
21    /// 扩展映射表（名称到类型的映射）
22    pub extension_registry: HashMap<String, ExtensionType>,
23    /// XML Schema 缓存
24    pub xml_schema_cache: HashMap<String, String>,
25    /// 预计算的哈希值（用于快速验证）
26    pub content_hash: u64,
27}

29/// 扩展类型枚举
30#[derive(Debug, Clone, Serialize, Deserialize)]
31pub enum ExtensionType {
32    Node { group: Option<String>, content_spec: String },
33    Mark { group: Option<String>, spanning: bool },
34}

36/// 可序列化的 Schema 结构
37#[derive(Debug, Clone, Serialize, Deserialize)]
38pub struct SerializableSchema {
39    pub nodes: HashMap<String, SerializableNodeType>,
40    pub marks: HashMap<String, SerializableMarkType>,
41    pub top_node: Option<String>,
42}
43
44#[derive(Debug, Clone, Serialize, Deserialize)]
45pub struct SerializableNodeType {
46    pub group: Option<String>,
47    pub content_spec: String,
48}
49
50#[derive(Debug, Clone, Serialize, Deserialize)]
51pub struct SerializableMarkType {
52    pub group: Option<String>,
53    pub spanning: bool,
54}
55
56impl SerializableSchema {
57    pub fn from_schema(schema: &mf_model::schema::Schema) -> Self {
58        let mut nodes = HashMap::new();
59        let mut marks = HashMap::new();
60
61        for (name, node_type) in &schema.nodes {
62            nodes.insert(
63                name.clone(),
64                SerializableNodeType {
65                    group: node_type.spec.group.clone(),
66                    content_spec: format!("{:?}", node_type.spec.content),
67                },
68            );
69        }
70
71        for (name, mark_type) in &schema.marks {
72            marks.insert(
73                name.clone(),
74                SerializableMarkType {
75                    group: mark_type.spec.group.clone(),
76                    spanning: mark_type.spec.spanning.unwrap_or(false),
77                },
78            );
79        }
80
81        Self { nodes, marks, top_node: schema.spec.top_node.clone() }
82    }
83}

129    /// 构建快照
130    pub fn build(self) -> ForgeResult<CoreSnapshot> {
131        let start_time = Instant::now();
132        println!("正在构建核心快照...");
133
134        // 1. 获取配置
135        let config = self.config.unwrap_or_default();
136
137        // 2. 构建扩展管理器
138        let mut extension_builder = ExtensionManager::builder();
139
140        // 添加 XML Schema
141        for path in &self.xml_schema_paths {
142            extension_builder = extension_builder.add_xml_file(path);
143        }
144
145        // 添加自定义扩展
146        extension_builder =
147            extension_builder.add_extensions(self.custom_extensions);
148
149        let extension_manager = extension_builder.build()?;
150        let schema = extension_manager.get_schema();
151
152        // 3. 序列化核心数据（使用 JSON 格式避免 Serialize trait 问题）
153        let schema_json =
154            serde_json::to_string(&SerializableSchema::from_schema(&schema))
155                .map_err(|e| {
156                    crate::error::error_utils::config_error(format!(
157                        "序列化 Schema 失败: {}",
158                        e
159                    ))
160                })?;
161
162        // 4. 构建扩展注册表
163        let mut extension_registry = HashMap::new();
164
165        for (name, node_type) in &schema.nodes {
166            extension_registry.insert(
167                name.clone(),
168                ExtensionType::Node {
169                    group: node_type.spec.group.clone(),
170                    content_spec: format!("{:?}", node_type.spec.content),
171                },
172            );
173        }
174
175        for (name, mark_type) in &schema.marks {
176            extension_registry.insert(
177                name.clone(),
178                ExtensionType::Mark {
179                    group: mark_type.spec.group.clone(),
180                    spanning: mark_type.spec.spanning.unwrap_or(false),
181                },
182            );
183        }
184
185        // 5. 缓存 XML Schema
186        let mut xml_schema_cache = HashMap::new();
187        for path in &self.xml_schema_paths {
188            if let Ok(content) = std::fs::read_to_string(path) {
189                xml_schema_cache.insert(path.clone(), content);
190            }
191        }
192
193        // 6. 计算内容哈希
194        let mut hasher = std::collections::hash_map::DefaultHasher::new();
195        use std::hash::{Hash, Hasher};
196        schema_json.hash(&mut hasher);
197        config.to_json().unwrap_or_default().hash(&mut hasher);
198        let content_hash = hasher.finish();
199
200        let snapshot = CoreSnapshot {
201            version: env!("CARGO_PKG_VERSION").to_string(),
202            created_at: std::time::SystemTime::now()
203                .duration_since(std::time::UNIX_EPOCH)
204                .unwrap()
205                .as_secs(),
206            config,
207            schema_json,
208            extension_registry,
209            xml_schema_cache,
210            content_hash,
211        };
212
213        println!("快照构建完成，耗时: {:?}", start_time.elapsed());
214        Ok(snapshot)
215    }
216}

218/// 快照管理器
219pub struct SnapshotManager;
220
221impl SnapshotManager {
222    /// 保存快照到文件
223    pub fn save_to_file(
224        snapshot: &CoreSnapshot,
225        path: &str,
226    ) -> ForgeResult<()> {
227        let data = bincode::serialize(snapshot).map_err(|e| {
228            crate::error::error_utils::config_error(format!(
229                "序列化快照失败: {}",
230                e
231            ))
232        })?;
233
234        std::fs::write(path, data).map_err(|e| {
235            crate::error::error_utils::config_error(format!(
236                "写入快照文件失败: {}",
237                e
238            ))
239        })?;
240
241        println!("快照已保存到: {}", path);
242        Ok(())
243    }
244
245    /// 从文件加载快照
246    pub fn load_from_file(path: &str) -> ForgeResult<CoreSnapshot> {
247        let data = std::fs::read(path).map_err(|e| {
248            crate::error::error_utils::config_error(format!(
249                "读取快照文件失败: {}",
250                e
251            ))
252        })?;
253
254        let snapshot: CoreSnapshot =
255            bincode::deserialize(&data).map_err(|e| {
256                crate::error::error_utils::config_error(format!(
257                    "反序列化快照失败: {}",
258                    e
259                ))
260            })?;
261
262        // 验证快照版本
263        if snapshot.version != env!("CARGO_PKG_VERSION") {
264            return Err(crate::error::error_utils::config_error(format!(
265                "快照版本不匹配: {} vs {}",
266                snapshot.version,
267                env!("CARGO_PKG_VERSION")
268            )));
269        }
270
271        Ok(snapshot)
272    }
273
274    /// 验证快照有效性
275    pub fn validate_snapshot(snapshot: &CoreSnapshot) -> ForgeResult<()> {
276        // 1. 检查版本兼容性
277        let current_version = env!("CARGO_PKG_VERSION");
278        if snapshot.version != current_version {
279            return Err(crate::error::error_utils::config_error(format!(
280                "快照版本 {} 与当前版本 {} 不兼容",
281                snapshot.version, current_version
282            )));
283        }
284
285        // 2. 检查数据完整性
286        if snapshot.schema_json.is_empty() {
287            return Err(crate::error::error_utils::config_error(
288                "快照中 Schema 数据为空".to_string(),
289            ));
290        }
291
292        // 3. 验证配置
293        snapshot.config.validate().map_err(|e| {
294            crate::error::error_utils::config_error(format!(
295                "快照配置无效: {}",
296                e
297            ))
298        })?;
299
300        Ok(())
301    }

303    /// 从快照恢复 ExtensionManager
304    pub fn restore_extension_manager(
305        snapshot: &CoreSnapshot
306    ) -> ForgeResult<ExtensionManager> {
307        let start_time = Instant::now();
308
309        // 从快照重建扩展管理器
310        let mut extensions = Vec::new();
311
312        // 从扩展注册表重建扩展
313        for (name, ext_type) in &snapshot.extension_registry {
314            match ext_type {
315                ExtensionType::Node { .. } => {
316                    // 创建基础节点扩展
317                    let node =
318                        crate::node::Node::create(name, Default::default());
319                    extensions.push(crate::types::Extensions::N(node));
320                },
321                ExtensionType::Mark { .. } => {
322                    // 创建基础标记扩展
323                    let mark = crate::mark::Mark::new(name, Default::default());
324                    extensions.push(crate::types::Extensions::M(mark));
325                },
326            }
327        }
328
329        // 使用扩展列表创建基础管理器
330        let manager = ExtensionManager::new(&extensions)?;
331
332        crate::metrics::snapshot_restore_duration(start_time.elapsed());
333        Ok(manager)
334    }

336    /// 检查快照是否需要更新
337    pub fn needs_update(
338        snapshot: &CoreSnapshot,
339        xml_paths: &[String],
340    ) -> bool {
341        // 检查 XML 文件是否发生变化
342        for path in xml_paths {
343            if let Ok(metadata) = std::fs::metadata(path) {
344                if let Ok(modified) = metadata.modified() {
345                    let modified_secs = modified
346                        .duration_since(std::time::UNIX_EPOCH)
347                        .unwrap_or_default()
348                        .as_secs();
349
350                    if modified_secs > snapshot.created_at {
351                        return true;
352                    }
353                }
354            }
355        }
356
357        false
358    }
