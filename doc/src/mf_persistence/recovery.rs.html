1//! 启动恢复：参考 price-storage 的做法（快照 + 事务重放）
2use std::sync::Arc;
3
4use crate::api::{EventStore};
5use crate::ser::{SnapshotData, TypeWrapper};
6use crate::step_factory::StepFactoryRegistry;
7
8/// 从存储恢复状态：加载最新快照，重放其后的事件
9pub async fn recover_state<E: EventStore + 'static>(
10    store: &E,
11    doc_id: &str,
12    configuration: &mf_state::Configuration,
13    step_factory: &StepFactoryRegistry,
14    batch: u32,
15) -> anyhow::Result<mf_state::State> {
16    // 1) 快照
17    let mut state = if let Some(snap) = store.latest_snapshot(doc_id).await? {
18        let bytes = zstd::decode_all(std::io::Cursor::new(snap.state_blob))?;
19        let snap_data: SnapshotData = serde_json::from_slice(&bytes)?;
20        let ser = mf_state::state::StateSerialize {
21            node_pool: snap_data.node_pool,
22            state_fields: snap_data.state_fields,
23        };
24        mf_state::State::deserialize(&ser, configuration).await?
25    } else {
26        // 当无快照时，从配置生成空状态
27        mf_state::State::new(Arc::new(configuration.clone()))?
28    };
29
30    // 2) 事件重放
31    let mut from_lsn =
32        store.latest_snapshot(doc_id).await?.map(|s| s.upto_lsn).unwrap_or(0);
33    loop {
34        let evs = store.load_since(doc_id, from_lsn, batch).await?;
35        if evs.is_empty() {
36            break;
37        }
38        for ev in evs {
39            let payload = zstd::decode_all(std::io::Cursor::new(ev.payload))?;
40            let frames: Vec<TypeWrapper> = serde_json::from_slice(&payload)?;
41            let mut tr = mf_state::Transaction::new(&state);
42            for f in frames {
43                tr.step(step_factory.create(&f.type_id, &f.data))?;
44            }
45            tr.commit();
46            state = state.apply(tr).await?.state;
47            from_lsn = ev.lsn;
48        }
49    }
50    Ok(state)
51}
